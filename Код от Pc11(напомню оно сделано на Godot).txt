# =============================================================
#  PC11 — Radchenko Windows 82 (сводная версия, преемник PC10)
#  Автор: Никита Радченко    GDScript-порт для Godot 4
#  Узел: Control (корневой)
#  Версия: 4.0  (DOS/CGA/VGA/TIGA/AdLib/x87 эмулятор, редакция Pc10 -> Pc11)
# =============================================================
#
#  Как подключить:
#    1. Создай сцену, корневой узел — Control (на весь экран).
#    2. Прикрепи этот скрипт к нему.
#    3. Запусти сцену — появится рабочий стол PC11.
#
#  Архитектура (база PC9/PC10, сохранена без изменений):
#    CPU8086      — процессор Intel 8086 (110+ команд + новые опкоды)
#    MemoryBus    — 1 МБ ОЗУ (PackedByteArray) с безопасным доступом
#    DOSKernel    — INT 21h / 20h / 10h / 11h / 12h / 15h / 16h / 1Ah
#    CGADisplay   — видеорежим 320×200, 4 цвета
#    AdLib/OPL2   — звуковая карта (FM-синтез, 9 каналов)
#    DOSLoader    — загрузчик .COM и .EXE (MZ-header + relocation)
#    KeyboardBuf  — буфер клавиш + BIOS INT 16h
#    SystemTimer  — таймер 18.2 Гц (BIOS 0x40:0x6C)
#    DesktopUI    — рабочий стол, панель задач, меню Пуск
#    PowerShell   — встроенный терминал
#
#  Что нового по сравнению с PC8 (унаследовано от PC9/PC10):
#    - INT 11h (опрос оборудования) и INT 12h (объём памяти)
#    - INT 15h (микрозадержки, джойстик, расширенная память)
#    - INT 1Ah (системные часы / генератор случайных чисел)
#    - Звуковая карта AdLib/OPL2 (порты 0x388/0x389) с реальным
#      FM-синтезом через AudioStreamGenerator
#    - Новые опкоды x86: PUSH imm8/imm16 (0x6A/0x68),
#      ENTER/LEAVE (0xC8/0xC9), AAA/DAS (0x37/0x2F)
#    - Безопасные функции доступа к памяти (защита от выхода
#      за пределы ОЗУ во всех новых блоках)
#    - Деление на ноль теперь вызывает настоящее INT 00h,
#      а не безусловный cpu_halt
#
#  Что было в PC10 (наследуется без изменений):
#    - VGATextDisplay        — текстовый видеорежим 80×25 (Mode 3h),
#                              16-цветная DOS-палитра, аппаратный курсор
#    - TIGADisplay/TIGACoprocessor — графический ускоритель 1280×1024,
#                              256 цветов, HLE BitBLT/линии/заливка,
#                              Host Interface Registers, Raster Ops
#    - ExtendedDOSKernel     — INT 20h/25h/26h/27h/2Fh/30h (абсолютный
#                              дисковый I/O, TSR, мультиплекс, CP/M anchor)
#    - EMS Manager (INT 67h) — Expanded Memory Manager, страницы 16КБ
#    - PC10_Advanced_Memory_Bus — 3 МБ физической памяти, линия A20,
#                              GDT-дескрипторы, защищённый режим (заглушка)
#    - PC10_IDE_Controller   — аппаратный ATA/IDE контроллер портов 0x1F0-0x1F7
#    - PC10_AccountManager   — система аккаунтов (SHA-256, JSON-база)
#    - PC10_Security_Bootloader — диалог подтверждения запуска .COM/.EXE
#    - Расширенный декодер опкодов 0x0F (Jcc rel16, SHLD, PUSH/POP FS/GS),
#      PUSHA/POPA, IMUL imm16/imm8, INSB/INSW/OUTSB/OUTSW, DAA/AAS
#
#  Что нового в PC11 (этот файл, добавлено сверху над базой PC10):
#    - Расширенная DOS-файловая подсистема (pc11_*) поверх старого
#      cpu_file_handles: CREATE (0x3C), полноценный OPEN с режимами
#      RDONLY/WRONLY/RDWR (0x3D), CLOSE (0x3E), READ (0x3F), WRITE
#      в произвольный handle (0x40), DELETE (0x41), LSEEK с SEEK_SET/
#      CUR/END и 32-битным смещением (0x42), GET/SET ATTRIB (0x43),
#      DUP/FORCE DUP хэндла (0x45/0x46), RENAME (0x56), GET/SET
#      FILETIME в DOS-формате (0x57), CREATETEMP со случайным именем
#      (0x5A)
#    - Пул хэндлов с переиспользованием освобождённых номеров вместо
#      безусловного наращивания счётчика, как было в базовом PC10
#    - Безопасная резолюция путей файлов (защита от обратных слешей
#      и абсолютных путей, автосоздание недостающих подкаталогов
#      в user://dos_c/)
#    - Хранение пути и режима открытия на каждый дескриптор —
#      необходимо для корректной работы DUP/RENAME/SEEK
#    - last_error как отдельное поле, имитирующее DOS extended
#      error code (INT 21h, AH=59h в реальном DOS)
#    - Мышь PS/2 (Microsoft Mouse API, INT 33h): инициализация,
#      показ/скрытие курсора, позиция и кнопки, программная
#      установка позиции, ограничивающая область (clip), счётчики
#      нажатий/отпусканий, пользовательский обработчик событий
#      (event handler) в стиле HLE, чувствительность мыши.
#      События мыши Godot (motion/button) транслируются прямо
#      в координаты и кнопки DOS-мыши.
#    - Последовательный порт COM1 (UART 8250/16450, INT 14h):
#      полный набор регистров 0x3F8-0x3FF (THR/RBR, IER, IIR/FCR,
#      LCR с поддержкой DLAB для делителя скорости, MCR, LSR, MSR),
#      программный loopback по умолчанию для тестирования
#      терминальных программ, лог отправленных байт, команды
#      PowerShell MOUSE STATUS / COM STATUS / COM ECHO.
#    - Настоящий BDA-кольцевой буфер клавиатуры (0x041A-0x043E)
#      вместо вспомогательного GDScript-массива: скан-код и ASCII
#      хранятся прямо в памяти эмулятора, указатели HEAD/TAIL —
#      по стандартным адресам IBM PC BIOS.
#    - Флаг cpu_halt_waiting_key — настоящее "зависание" CPU на
#      INT 16h AH=00h без лишних тактов на повторный fetch/decode.
#    - Визуальный курсор мыши PS/2, отрисовываемый поверх VGA-
#      текстового экрана (раньше координаты мыши существовали
#      только в памяти, без отображения на экране).
#    - Математический сопроцессор x87 (опкоды 0xD8-0xDF): полный
#      стек ST(0)-ST(7), FADD/FSUB/FMUL/FDIV в прямой и реверсивной
#      форме, FCOM/FCOMP/FCOMPP со Status Word (C0/C2/C3), FLD/FST/
#      FSTP для 32- и 64-битных IEEE 754 значений из памяти DOS-
#      программы, FCHS/FABS/FSQRT, константы FLDZ/FLD1/FLDPI,
#      FXCH, FLDCW/FSTCW, полноценные знаковые FILD/FIST/FISTP
#      для 16- и 32-битных целых чисел (m16int/m32int), а также
#      критическая синхронизация FPU<->CPU через FSTSW AX (DF E0),
#      позволяющая программе проверять флаги C0-C3 обычными
#      командами TEST/JZ/JNZ основного процессора.
#      Команды PowerShell FPU STATUS / FPU RESET для диагностики.
# =============================================================

# ══════════════════════════════════════════════════════════════
#  БЛОК 1 — ПАМЯТЬ (с безопасными функциями доступа)
# ══════════════════════════════════════════════════════════════
var memory : PackedByteArray

func mem_init() -> void:
	memory = PackedByteArray()
	memory.resize(RAM_SIZE)
	memory.fill(0)

func mem_rb(seg: int, off: int) -> int:
	var addr : int = ((seg << 4) + (off & 0xFFFF)) & 0xFFFFF
	return memory[addr]

func mem_wb(seg: int, off: int, val: int) -> void:
	var addr : int = ((seg << 4) + (off & 0xFFFF)) & 0xFFFFF
	memory[addr] = val & 0xFF

func mem_rw(seg: int, off: int) -> int:
	var addr : int = ((seg << 4) + (off & 0xFFFF)) & 0xFFFFF
	var addr2 : int = ((seg << 4) + ((off + 1) & 0xFFFF)) & 0xFFFFF
	return memory[addr] | (memory[addr2] << 8)

func mem_ww(seg: int, off: int, val: int) -> void:
	var addr : int = ((seg << 4) + (off & 0xFFFF)) & 0xFFFFF
	var addr2 : int = ((seg << 4) + ((off + 1) & 0xFFFF)) & 0xFFFFF
	memory[addr]  = val & 0xFF
	memory[addr2] = (val >> 8) & 0xFF

func mem_get_phys(seg_val: int, off: int) -> int:
	return ((seg_val << 4) + (off & 0xFFFF)) & 0xFFFFF

# Безопасная запись/чтение по уже посчитанному физическому адресу
# (используется новым кодом загрузчика и подсистемами PC9, защищает
# от выхода за пределы массива memory при любых вычислениях адреса)
func safe_write_8(phys_addr: int, val: int) -> void:
	var addr : int = phys_addr & 0xFFFFF
	if addr < memory.size():
		memory[addr] = val & 0xFF

func safe_write_16(phys_addr: int, val: int) -> void:
	safe_write_8(phys_addr, val & 0xFF)
	safe_write_8(phys_addr + 1, (val >> 8) & 0xFF)

func safe_read_8(phys_addr: int) -> int:
	var addr : int = phys_addr & 0xFFFFF
	if addr < memory.size():
		return memory[addr]
	return 0

func safe_read_16(phys_addr: int) -> int:
	var lo : int = safe_read_8(phys_addr)
	var hi : int = safe_read_8(phys_addr + 1)
	return lo | (hi << 8)

# ══════════════════════════════════════════════════════════════
#  БЛОК 1B — PC10: РАСШИРЕННАЯ ШИНА ПАМЯТИ (3 МБ, A20, GDT, EMS)
#  Архитектура: 24-битный физический адрес (до 16 МБ), квирк линии A20,
#               парсер дескрипторов GDT, EMS-страницы по INT 67h.
#  Это самостоятельный слой НАД основной 1 МБ памятью PC9 — используется
#  только новым кодом PC10 (XMS/EMS/Protected-mode заглушки), основной
#  CPU8086 продолжает работать через memory/mem_rb/mem_wb как раньше.
# ══════════════════════════════════════════════════════════════
var pc10_ext_ram          : PackedByteArray
var pc10_a20_enabled      : bool = false
var pc10_protected_mode   : bool = false
var pc10_gdtr_base        : int  = 0x00000000
var pc10_gdtr_limit       : int  = 0x0000

class PC10_SegmentDescriptor:
	var base_address : int  = 0
	var limit        : int  = 0
	var is_writable  : bool = true
	var is_code      : bool = false
	var present      : bool = false

func pc10_ext_mem_init() -> void:
	pc10_ext_ram = PackedByteArray()
	pc10_ext_ram.resize(PC10_EXT_RAM_SIZE)
	pc10_ext_ram.fill(0x00)
	print("[PC10 Hardware] Инициализировано %d байт честной физической памяти (3 МБ)." % PC10_EXT_RAM_SIZE)

func pc10_ext_read_byte(selector_or_segment: int, offset: int) -> int:
	var phys_addr : int = 0
	if not pc10_protected_mode:
		phys_addr = (selector_or_segment << 4) + offset
		if not pc10_a20_enabled:
			phys_addr = phys_addr & 0xFFFFF
	else:
		var desc : PC10_SegmentDescriptor = pc10_parse_gdt_descriptor(selector_or_segment)
		if not desc.present:
			pc10_trigger_hardware_fault(11, selector_or_segment)
			return 0
		if offset > desc.limit:
			pc10_trigger_hardware_fault(13, offset)
			return 0
		phys_addr = desc.base_address + offset
	if phys_addr >= PC10_EXT_RAM_SIZE:
		pc10_trigger_hardware_fault(13, phys_addr)
		return 0
	return pc10_ext_ram[phys_addr]

func pc10_ext_write_byte(selector_or_segment: int, offset: int, value: int) -> void:
	var phys_addr : int = 0
	if not pc10_protected_mode:
		phys_addr = (selector_or_segment << 4) + offset
		if not pc10_a20_enabled:
			phys_addr = phys_addr & 0xFFFFF
	else:
		var desc : PC10_SegmentDescriptor = pc10_parse_gdt_descriptor(selector_or_segment)
		if not desc.present:
			pc10_trigger_hardware_fault(11, selector_or_segment)
			return
		if offset > desc.limit or not desc.is_writable:
			pc10_trigger_hardware_fault(13, offset)
			return
		phys_addr = desc.base_address + offset
	if phys_addr >= PC10_EXT_RAM_SIZE:
		pc10_trigger_hardware_fault(13, phys_addr)
		return
	pc10_ext_ram[phys_addr] = value & 0xFF

func pc10_ext_read_word(selector_or_segment: int, offset: int) -> int:
	var lo : int = pc10_ext_read_byte(selector_or_segment, offset)
	var hi : int = pc10_ext_read_byte(selector_or_segment, offset + 1)
	return lo | (hi << 8)

func pc10_ext_write_word(selector_or_segment: int, offset: int, value: int) -> void:
	pc10_ext_write_byte(selector_or_segment, offset, value & 0xFF)
	pc10_ext_write_byte(selector_or_segment, offset + 1, (value >> 8) & 0xFF)

func pc10_parse_gdt_descriptor(selector: int) -> PC10_SegmentDescriptor:
	var desc : PC10_SegmentDescriptor = PC10_SegmentDescriptor.new()
	var gdt_index : int = (selector >> 3) * 8
	var descriptor_phys_addr : int = pc10_gdtr_base + gdt_index
	if gdt_index + 7 > pc10_gdtr_limit:
		return desc
	if descriptor_phys_addr + 7 >= pc10_ext_ram.size():
		return desc
	var b0 : int = pc10_ext_ram[descriptor_phys_addr]
	var b1 : int = pc10_ext_ram[descriptor_phys_addr + 1]
	var b2 : int = pc10_ext_ram[descriptor_phys_addr + 2]
	var b3 : int = pc10_ext_ram[descriptor_phys_addr + 3]
	var b4 : int = pc10_ext_ram[descriptor_phys_addr + 4]
	var b5 : int = pc10_ext_ram[descriptor_phys_addr + 5]
	var b7 : int = pc10_ext_ram[descriptor_phys_addr + 7]
	desc.limit = b0 | (b1 << 8)
	desc.base_address = b2 | (b3 << 8) | (b4 << 16)
	desc.base_address |= (b7 << 24)
	desc.present     = (b5 & 0x80) != 0
	desc.is_code     = (b5 & 0x08) != 0
	desc.is_writable = ((b5 & 0x02) != 0) if not desc.is_code else false
	return desc

func pc10_trigger_hardware_fault(vector: int, info: int) -> void:
	_ps_print("[PC10 FAULT] Аппаратное исключение #%d (доп. данные: 0x%X)\n" % [vector, info])

# ── INT 15h, AH=87h — перемещение блока расширенной памяти через GDT ──
func pc10_handle_int15_move_ext_memory() -> void:
	var num_words : int = cpu_regs["CX"]
	var num_bytes : int = num_words * 2
	var es : int = cpu_segs["ES"]
	var si : int = cpu_regs["SI"]
	var gdt_table_phys : int = (es << 4) + si

	var src_desc_addr : int = gdt_table_phys + 0x08
	var dst_desc_addr : int = gdt_table_phys + 0x10

	if src_desc_addr + 7 >= RAM_SIZE or dst_desc_addr + 7 >= RAM_SIZE:
		cpu_flags["CF"] = 1
		return

	var src_base : int = memory[src_desc_addr+2] | (memory[src_desc_addr+3] << 8) | (memory[src_desc_addr+4] << 16) | (memory[src_desc_addr+7] << 24)
	var dst_base : int = memory[dst_desc_addr+2] | (memory[dst_desc_addr+3] << 8) | (memory[dst_desc_addr+4] << 16) | (memory[dst_desc_addr+7] << 24)

	if src_base + num_bytes > PC10_EXT_RAM_SIZE or dst_base + num_bytes > PC10_EXT_RAM_SIZE:
		cpu_regs["AX"] = 0x0300
		cpu_flags["CF"] = 1
		_ps_print("[BIOS INT 15h] Критическая ошибка: выход за лимит 3 МБ ОЗУ!\n")
		return

	for i in range(num_bytes):
		if src_base + i < pc10_ext_ram.size() and dst_base + i < pc10_ext_ram.size():
			pc10_ext_ram[dst_base + i] = pc10_ext_ram[src_base + i]

	cpu_flags["CF"] = 0
	cpu_regs["AX"] = 0x0000

# ── EMS Manager — INT 67h (Expanded Memory Manager) ───────────────────
var pc10_ems_handle_allocated : Dictionary = {}   # handle:int -> Array[int] (логические страницы)

	pc10_ems_expanded_memory.resize(EMS_TOTAL_PAGES * EMS_PAGE_SIZE)
	pc10_ems_expanded_memory.fill(0)
	pc10_ems_handle_allocated.clear()
	pc10_ems_page_mapping = [-1, -1, -1, -1]

func pc10_ems_get_allocated_count() -> int:
	var count : int = 0
	for handle in pc10_ems_handle_allocated:
		var arr : Array = pc10_ems_handle_allocated[handle]
		count += arr.size()
	return count

func pc10_ems_find_free_pages(count: int) -> Array:
	var free_list : Array = []
	var used_set : Dictionary = {}
	for handle in pc10_ems_handle_allocated:
		for page in pc10_ems_handle_allocated[handle]:
			used_set[page] = true
	for i in range(EMS_TOTAL_PAGES):
		if not used_set.has(i):
			free_list.append(i)
			if free_list.size() == count:
				return free_list
	return []

func _int_67h_ems() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	match ah:
		0x40:
			cpu_regs["AX"] = 0x0000
			_ps_print("[EMM] Проверка статуса: EMS драйвер PC10 активен.\n")
		0x41:
			cpu_regs["BX"] = EMS_FRAME_SEGMENT
			cpu_regs["AX"] = 0x0000
		0x42:
			cpu_regs["BX"] = EMS_TOTAL_PAGES - pc10_ems_get_allocated_count()
			cpu_regs["DX"] = EMS_TOTAL_PAGES
			cpu_regs["AX"] = 0x0000
		0x43:
			var pages_requested : int = cpu_regs["BX"]
			var new_handle : int = pc10_ems_handle_allocated.keys().size() + 1
			if pages_requested <= (EMS_TOTAL_PAGES - pc10_ems_get_allocated_count()):
				var page_indices : Array = pc10_ems_find_free_pages(pages_requested)
				pc10_ems_handle_allocated[new_handle] = page_indices
				cpu_regs["DX"] = new_handle
				cpu_regs["AX"] = 0x0000
				_ps_print("[EMM] Выделено %d страниц под Handle %d\n" % [pages_requested, new_handle])
			else:
				cpu_regs["AX"] = 0x8800
		0x44:
			var physical_window : int = cpu_regs["AX"] & 0xFF
			var logical_page_offset : int = cpu_regs["BX"]
			var handle : int = cpu_regs["DX"]
			if pc10_ems_handle_allocated.has(handle) and physical_window >= 0 and physical_window < 4:
				var allocated_pages : Array = pc10_ems_handle_allocated[handle]
				if logical_page_offset < allocated_pages.size():
					pc10_ems_page_mapping[physical_window] = allocated_pages[logical_page_offset]
					cpu_regs["AX"] = 0x0000
				else:
					cpu_regs["AX"] = 0x8A00
			else:
				cpu_regs["AX"] = 0x8900
		0x45:
			var handle2 : int = cpu_regs["DX"]
			if pc10_ems_handle_allocated.has(handle2):
				pc10_ems_handle_allocated.erase(handle2)
				for i in range(4): pc10_ems_page_mapping[i] = -1
				cpu_regs["AX"] = 0x0000
				_ps_print("[EMM] Память Handle %d успешно освобождена.\n" % handle2)
			else:
				cpu_regs["AX"] = 0x8900
		_:
			cpu_regs["AX"] = 0x0000   # неизвестная подфункция

# ── Fetch ──────────────────────────────────────────────────────
func cpu_fetch_b() -> int:
	var v : int = mem_rb(cpu_segs["CS"], cpu_ip)
	cpu_ip = (cpu_ip + 1) & 0xFFFF
	return v

func cpu_fetch_w() -> int:
	var lo : int = cpu_fetch_b()
	var hi : int = cpu_fetch_b()
	return lo | (hi << 8)

func cpu_fetch_sb() -> int:
	var v : int = cpu_fetch_b()
	return v - 256 if v & 0x80 else v

func cpu_fetch_sw() -> int:
	var v : int = cpu_fetch_w()
	return v - 65536 if v & 0x8000 else v

# ── Регистры 8-бит ────────────────────────────────────────────
func cpu_get_r8(idx: int) -> int:
	var reg : String = REG16[idx % 4]
	if idx >= 4:
		return (cpu_regs[reg] >> 8) & 0xFF
	return cpu_regs[reg] & 0xFF

func cpu_set_r8(idx: int, val: int) -> void:
	var reg : String = REG16[idx % 4]
	if idx >= 4:
		cpu_regs[reg] = (cpu_regs[reg] & 0x00FF) | ((val & 0xFF) << 8)
	else:
		cpu_regs[reg] = (cpu_regs[reg] & 0xFF00) | (val & 0xFF)

# ── Адресация с учётом seg_override ───────────────────────────
func cpu_get_phys(seg_name: String, off: int) -> int:
	var seg_val : int
	if cpu_seg_override >= 0 and seg_name == "DS":
		seg_val = cpu_seg_override
	else:
		seg_val = cpu_segs[seg_name]
	return ((seg_val << 4) + (off & 0xFFFF)) & 0xFFFFF

func cpu_rb(seg: String, off: int) -> int:
	return memory[cpu_get_phys(seg, off)]

func cpu_wb(seg: String, off: int, val: int) -> void:
	memory[cpu_get_phys(seg, off)] = val & 0xFF

func cpu_rw(seg: String, off: int) -> int:
	var a1 : int = cpu_get_phys(seg, off)
	var a2 : int = cpu_get_phys(seg, (off + 1) & 0xFFFF)
	return memory[a1] | (memory[a2] << 8)

func cpu_ww(seg: String, off: int, val: int) -> void:
	var a1 : int = cpu_get_phys(seg, off)
	var a2 : int = cpu_get_phys(seg, (off + 1) & 0xFFFF)
	memory[a1] = val & 0xFF
	memory[a2] = (val >> 8) & 0xFF

# ── Флаги ─────────────────────────────────────────────────────
func cpu_upd_flags(res: int, size: int = 16, sub: bool = false, a: int = 0, b: int = 0) -> void:
	var mask : int = 0xFFFF if size == 16 else 0xFF
	var sign : int = 0x8000 if size == 16 else 0x80
	var r    : int = res & mask
	cpu_flags["ZF"] = 1 if r == 0 else 0
	cpu_flags["SF"] = 1 if (r & sign) != 0 else 0
	var bits : int = r & 0xFF
	var cnt  : int = 0
	while bits:
		cnt += bits & 1
		bits >>= 1
	cpu_flags["PF"] = 1 if cnt % 2 == 0 else 0
	if sub:
		cpu_flags["CF"] = 1 if a < b else 0
		cpu_flags["OF"] = 1 if ((a ^ b) & sign) != 0 and ((a ^ r) & sign) != 0 else 0
		cpu_flags["AF"] = 1 if (a & 0xF) < (b & 0xF) else 0
	else:
		cpu_flags["CF"] = 1 if res > mask else 0
		cpu_flags["OF"] = 1 if ((a ^ b) & sign) == 0 and ((a ^ r) & sign) != 0 else 0
		cpu_flags["AF"] = 1 if ((a & 0xF) + (b & 0xF)) > 0xF else 0

# ── Стек ──────────────────────────────────────────────────────
func cpu_push(val: int) -> void:
	cpu_regs["SP"] = (cpu_regs["SP"] - 2) & 0xFFFF
	cpu_ww("SS", cpu_regs["SP"], val)

func cpu_pop() -> int:
	var v : int = cpu_rw("SS", cpu_regs["SP"])
	cpu_regs["SP"] = (cpu_regs["SP"] + 2) & 0xFFFF
	return v

# ── Декодер ModR/M ────────────────────────────────────────────
func cpu_decode_modrm(word: bool = true) -> Dictionary:
	var byte : int = cpu_fetch_b()
	var mod  : int = (byte >> 6) & 3
	var reg  : int = (byte >> 3) & 7
	var rm   : int =  byte       & 7

	if mod == 3:
		return {"mem": false, "reg": reg, "rm": rm, "word": word}

	var base : int = 0
	var seg  : String = "DS"

	match rm:
		0: base = cpu_regs["BX"] + cpu_regs["SI"]
		1: base = cpu_regs["BX"] + cpu_regs["DI"]
		2: base = cpu_regs["BP"] + cpu_regs["SI"]; seg = "SS"
		3: base = cpu_regs["BP"] + cpu_regs["DI"]; seg = "SS"
		4: base = cpu_regs["SI"]
		5: base = cpu_regs["DI"]
		6:
			if mod == 0:
				base = cpu_fetch_w()
			else:
				base = cpu_regs["BP"]; seg = "SS"
		7: base = cpu_regs["BX"]

	var disp : int = 0
	if   mod == 1: disp = cpu_fetch_sb()
	elif mod == 2: disp = cpu_fetch_w()

	var off : int = (base + disp) & 0xFFFF
	return {"mem": true, "reg": reg, "seg": seg, "off": off, "word": word}

func cpu_get_rm(m: Dictionary) -> int:
	if not m["mem"]:
		if m["word"]: return cpu_regs[REG16[m["rm"]]]
		else:         return cpu_get_r8(m["rm"])
	if m["word"]: return cpu_rw(m["seg"], m["off"])
	return cpu_rb(m["seg"], m["off"])

func cpu_set_rm(m: Dictionary, val: int) -> void:
	if not m["mem"]:
		if m["word"]: cpu_regs[REG16[m["rm"]]] = val & 0xFFFF
		else:         cpu_set_r8(m["rm"], val)
	else:
		if m["word"]: cpu_ww(m["seg"], m["off"], val)
		else:         cpu_wb(m["seg"], m["off"], val)

func cpu_get_reg(m: Dictionary) -> int:
	if m["word"]: return cpu_regs[REG16[m["reg"]]]
	return cpu_get_r8(m["reg"])

func cpu_set_reg(m: Dictionary, val: int) -> void:
	if m["word"]: cpu_regs[REG16[m["reg"]]] = val & 0xFFFF
	else:         cpu_set_r8(m["reg"], val)

# ══════════════════════════════════════════════════════════════
#  PC11 — МАТЕМАТИЧЕСКИЙ СОПРОЦЕССОР x87 (опкоды 0xD8-0xDF)
#  Полноценная эмуляция стека ST(0)-ST(7), как у реального 8087/
#  80287/80387: FADD/FSUB/FMUL/FDIV в прямой и реверсивной форме,
#  FLD/FST/FSTP для 32-битных и 64-битных IEEE 754 значений прямо
#  из памяти DOS-программы, FCOM/FCOMP/FCOMPP для сравнений,
#  FCHS/FABS/FSQRT, константы FLDZ/FLD1/FLDPI, FXCH, FLDCW/FSTCW.
# ══════════════════════════════════════════════════════════════

func fpu_push(val : float) -> void:
	fpu_st_top = (fpu_st_top - 1) & 7
	fpu_stack[fpu_st_top] = val

func fpu_pop() -> float:
	var val : float = fpu_stack[fpu_st_top]
	fpu_st_top = (fpu_st_top + 1) & 7
	return val

func fpu_get_st(i : int) -> float:
	return fpu_stack[(fpu_st_top + i) & 7]

func fpu_set_st(i : int, val : float) -> void:
	fpu_stack[(fpu_st_top + i) & 7] = val

# Флаги Status Word (C0, C1, C2, C3) — используются условными переходами FPU (FSTSW/SAHF)
func fpu_set_flags(c0 : int, c2 : int, c3 : int) -> void:
	fpu_status_word = (fpu_status_word & 0xBEFF) | (c0 << 8) | (c2 << 10) | (c3 << 14)

func fpu_reset() -> void:
	for i in range(8):
		fpu_stack[i] = 0.0
	fpu_st_top = 0
	fpu_status_word = 0x0000
	fpu_control_word = 0x037F

# ── Работа с памятью FPU (32-bit и 64-bit IEEE 754) ────────────
func fpu_read_mem32(phys_addr : int) -> float:
	var tmp : PackedByteArray = PackedByteArray([0, 0, 0, 0])
	for i in range(4): tmp[i] = memory[(phys_addr + i) & 0xFFFFF]
	return tmp.decode_float(0)

func fpu_read_mem64(phys_addr : int) -> float:
	var tmp : PackedByteArray = PackedByteArray([0, 0, 0, 0, 0, 0, 0, 0])
	for i in range(8): tmp[i] = memory[(phys_addr + i) & 0xFFFFF]
	return tmp.decode_double(0)

func fpu_write_mem32(phys_addr : int, val : float) -> void:
	var tmp : PackedByteArray = PackedByteArray([0, 0, 0, 0])
	tmp.encode_float(0, val)
	for i in range(4): memory[(phys_addr + i) & 0xFFFFF] = tmp[i]

func fpu_write_mem64(phys_addr : int, val : float) -> void:
	var tmp : PackedByteArray = PackedByteArray([0, 0, 0, 0, 0, 0, 0, 0])
	tmp.encode_double(0, val)
	for i in range(8): memory[(phys_addr + i) & 0xFFFFF] = tmp[i]

# ── Конвертация целых чисел (Integer Load/Store для FPU) ───────
func fpu_read_int16(phys_addr : int) -> float:
	var val : int = memory[phys_addr & 0xFFFFF] | (memory[(phys_addr + 1) & 0xFFFFF] << 8)
	if val & 0x8000: val -= 0x10000
	return float(val)

func fpu_write_int16(phys_addr : int, val : float) -> void:
	var int_val : int = int(round(val))
	memory[phys_addr & 0xFFFFF]       = int_val & 0xFF
	memory[(phys_addr + 1) & 0xFFFFF] = (int_val >> 8) & 0xFF

func fpu_read_int32(phys_addr : int) -> float:
	var val : int = (memory[phys_addr & 0xFFFFF]
		| (memory[(phys_addr + 1) & 0xFFFFF] << 8)
		| (memory[(phys_addr + 2) & 0xFFFFF] << 16)
		| (memory[(phys_addr + 3) & 0xFFFFF] << 24))
	if val & 0x80000000: val -= 0x100000000
	return float(val)

func fpu_write_int32(phys_addr : int, val : float) -> void:
	var int_val : int = int(round(val))
	memory[phys_addr & 0xFFFFF]       = int_val & 0xFF
	memory[(phys_addr + 1) & 0xFFFFF] = (int_val >> 8)  & 0xFF
	memory[(phys_addr + 2) & 0xFFFFF] = (int_val >> 16) & 0xFF
	memory[(phys_addr + 3) & 0xFFFFF] = (int_val >> 24) & 0xFF

# Хелпер для сравнений (FCOM/FCOMP/FCOMPP) — выставляет C0/C2/C3 в Status Word
func _fpu_com(val : float) -> void:
	var st0 : float = fpu_get_st(0)
	if st0 > val:
		fpu_set_flags(0, 0, 0)
	elif st0 < val:
		fpu_set_flags(0, 0, 1)
	elif st0 == val:
		fpu_set_flags(1, 0, 0)
	else:   # NaN или несопоставимо
		fpu_set_flags(1, 1, 1)

# ── Декодер инструкций x87 (опкоды 0xD8 - 0xDF) ───────────────
func cpu_exec_fpu(op : int) -> void:
	var m : Dictionary = cpu_decode_modrm(false)   # ModR/M у FPU всегда читается как 8-битный режим адресации
	var is_mem : bool = m["mem"]
	var reg_op : int  = m["reg"]                    # 3 бита, определяющие тип операции
	var rm     : int  = m["rm"]

	var phys_addr : int = 0
	if is_mem: phys_addr = cpu_get_phys(m["seg"], m["off"])

	match op:
		0xD8:   # D8: FADD/FMUL/FCOM/FSUB/FDIV с f32 (ОЗУ) или ST(0), ST(i)
			if is_mem:
				var v32 : float = fpu_read_mem32(phys_addr)
				match reg_op:
					0: fpu_set_st(0, fpu_get_st(0) + v32)   # FADD m32real
					1: fpu_set_st(0, fpu_get_st(0) * v32)   # FMUL m32real
					2: _fpu_com(v32)                        # FCOM m32real
					3: _fpu_com(v32); fpu_pop()              # FCOMP m32real
					4: fpu_set_st(0, fpu_get_st(0) - v32)   # FSUB m32real
					5: fpu_set_st(0, v32 - fpu_get_st(0))   # FSUBR m32real
					6: fpu_set_st(0, fpu_get_st(0) / v32)   # FDIV m32real
					7: fpu_set_st(0, v32 / fpu_get_st(0))   # FDIVR m32real
			else:
				var sti : float = fpu_get_st(rm)
				match reg_op:
					0: fpu_set_st(0, fpu_get_st(0) + sti)   # FADD ST(0), ST(i)
					1: fpu_set_st(0, fpu_get_st(0) * sti)   # FMUL ST(0), ST(i)
					2: _fpu_com(sti)                        # FCOM ST(0), ST(i)
					3: _fpu_com(sti); fpu_pop()              # FCOMP ST(0), ST(i)
					4: fpu_set_st(0, fpu_get_st(0) - sti)   # FSUB ST(0), ST(i)
					5: fpu_set_st(0, sti - fpu_get_st(0))   # FSUBR ST(0), ST(i)
					6: fpu_set_st(0, fpu_get_st(0) / sti)   # FDIV ST(0), ST(i)
					7: fpu_set_st(0, sti / fpu_get_st(0))   # FDIVR ST(0), ST(i)

		0xD9:   # D9: FLD/FST/Константы/Унарные операции
			if is_mem:
				match reg_op:
					0: fpu_push(fpu_read_mem32(phys_addr))          # FLD m32real
					2: fpu_write_mem32(phys_addr, fpu_get_st(0))    # FST m32real
					3: fpu_write_mem32(phys_addr, fpu_pop())        # FSTP m32real
					5: cpu_ww("DS", m["off"], fpu_control_word)     # FLDCW
					7: fpu_control_word = cpu_rw("DS", m["off"])    # FSTCW
			else:
				match reg_op:
					0: fpu_push(fpu_get_st(rm))   # FLD ST(i)
					1:   # FXCH ST(i)
						var tmp : float = fpu_get_st(0)
						fpu_set_st(0, fpu_get_st(rm))
						fpu_set_st(rm, tmp)
					4:   # Унарные операции и константы (D9 E0 - D9 FF)
						match rm:
							0: fpu_set_st(0, -fpu_get_st(0))          # FCHS (смена знака)
							1: fpu_set_st(0, absf(fpu_get_st(0)))     # FABS (модуль)
							4: fpu_push(0.0)                          # FLDZ
							5: fpu_push(1.0)                          # FLD1
							6: fpu_push(PI)                           # FLDPI
							_: fpu_set_st(0, sqrt(fpu_get_st(0)))     # FSQRT (упрощённый fallback)

		0xDB:   # DB: работа с 32-битными целыми числами (FILD/FIST/FISTP m32int)
			if is_mem:
				match reg_op:
					0: fpu_push(fpu_read_int32(phys_addr))          # FILD m32int
					2: fpu_write_int32(phys_addr, fpu_get_st(0))    # FIST m32int
					3: fpu_write_int32(phys_addr, fpu_pop())        # FISTP m32int
			else:
				if reg_op == 4:   # FNENI (Enable Interrupts) — в HLE-реализации игнорируется
					pass

		0xDC:   # DC: арифметика с m64real или ST(i), ST(0)
			if is_mem:
				var v64 : float = fpu_read_mem64(phys_addr)
				match reg_op:
					0: fpu_set_st(0, fpu_get_st(0) + v64)   # FADD m64real
					1: fpu_set_st(0, fpu_get_st(0) * v64)   # FMUL m64real
					2: _fpu_com(v64)                        # FCOM m64real
					3: _fpu_com(v64); fpu_pop()              # FCOMP m64real
					4: fpu_set_st(0, fpu_get_st(0) - v64)   # FSUB m64real
					5: fpu_set_st(0, v64 - fpu_get_st(0))   # FSUBR m64real
					6: fpu_set_st(0, fpu_get_st(0) / v64)   # FDIV m64real
					7: fpu_set_st(0, v64 / fpu_get_st(0))   # FDIVR m64real
			else:
				var st0_dc : float = fpu_get_st(0)
				match reg_op:
					0: fpu_set_st(rm, fpu_get_st(rm) + st0_dc)   # FADD ST(i), ST(0)
					1: fpu_set_st(rm, fpu_get_st(rm) * st0_dc)   # FMUL ST(i), ST(0)
					4: fpu_set_st(rm, fpu_get_st(rm) - st0_dc)   # FSUBR ST(i), ST(0) (реверс в x87)
					5: fpu_set_st(rm, st0_dc - fpu_get_st(rm))   # FSUB ST(i), ST(0)
					6: fpu_set_st(rm, fpu_get_st(rm) / st0_dc)   # FDIVR ST(i), ST(0)
					7: fpu_set_st(rm, st0_dc / fpu_get_st(rm))   # FDIV ST(i), ST(0)

		0xDD:   # DD: загрузка/сохранение 64-bit
			if is_mem:
				match reg_op:
					0: fpu_push(fpu_read_mem64(phys_addr))          # FLD m64real
					2: fpu_write_mem64(phys_addr, fpu_get_st(0))    # FST m64real
					3: fpu_write_mem64(phys_addr, fpu_pop())        # FSTP m64real
			else:
				match reg_op:
					2: fpu_set_st(rm, fpu_get_st(0))                # FST ST(i)
					3: fpu_set_st(rm, fpu_get_st(0)); fpu_pop()      # FSTP ST(i)

		0xDE:   # DE: арифметика ST(i), ST(0) с POP (FADDP, FMULP и т.д.)
			if not is_mem:
				var st0_de : float = fpu_get_st(0)
				match reg_op:
					0: fpu_set_st(rm, fpu_get_st(rm) + st0_de); fpu_pop()   # FADDP
					1: fpu_set_st(rm, fpu_get_st(rm) * st0_de); fpu_pop()   # FMULP
					3: _fpu_com(fpu_get_st(rm)); fpu_pop(); fpu_pop()       # FCOMPP (DE D9)
					4: fpu_set_st(rm, fpu_get_st(rm) - st0_de); fpu_pop()   # FSUBRP
					5: fpu_set_st(rm, st0_de - fpu_get_st(rm)); fpu_pop()   # FSUBP
					6: fpu_set_st(rm, fpu_get_st(rm) / st0_de); fpu_pop()   # FDIVRP
					7: fpu_set_st(rm, st0_de / fpu_get_st(rm)); fpu_pop()   # FDIVP

		0xDF:   # DF: работа с 16-битными целыми числами и синхронизация FPU<->CPU
			if is_mem:
				match reg_op:
					0: fpu_push(fpu_read_int16(phys_addr))          # FILD m16int
					2: fpu_write_int16(phys_addr, fpu_get_st(0))    # FIST m16int
					3: fpu_write_int16(phys_addr, fpu_pop())        # FISTP m16int
					# Опкоды 5 и 7 (FILD/FISTP m64int) используются в DOS крайне редко — оставлены как заглушки
			else:
				# ── Самый важный блок: синхронизация FPU и CPU ──
				if reg_op == 4 and rm == 0:
					# Команда FSTSW AX (опкод DF E0)
					# Копирует слово состояния FPU в регистр AX основного процессора,
					# чтобы программа могла проверить флаги C0-C3 обычными командами
					# CPU (TEST/JZ/JNZ) без отдельной инструкции перехода FPU.
					cpu_regs["AX"] = fpu_status_word

		_:
			_ps_print("[FPU] Нереализованный x87 опкод: 0x%02X\n" % op)

# ── ALU ───────────────────────────────────────────────────────
func cpu_alu(op_id: int, dst: int, src: int, size: int = 16) -> int:
	var r    : int = 0
	var mask : int = 0xFFFF if size == 16 else 0xFF
	match op_id:
		0: r = dst + src;           cpu_upd_flags(r, size, false, dst, src)  # ADD
		1:                                                                      # OR
			r = dst | src
			cpu_upd_flags(r, size)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		2: r = dst + src + cpu_flags["CF"]; cpu_upd_flags(r, size, false, dst, src)  # ADC
		3: r = dst - src - cpu_flags["CF"]; cpu_upd_flags(r, size, true,  dst, src)  # SBB
		4:                                                                      # AND
			r = dst & src
			cpu_upd_flags(r, size)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		5: r = dst - src; cpu_upd_flags(r, size, true, dst, src)              # SUB
		6:                                                                      # XOR
			r = dst ^ src
			cpu_upd_flags(r, size)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		7:                                                                      # CMP
			r = dst - src
			cpu_upd_flags(r, size, true, dst, src)
			return dst   # не пишем результат
	return r & mask

# ── Строковые команды ─────────────────────────────────────────
func cpu_stosb() -> void:
	cpu_wb("ES", cpu_regs["DI"], cpu_regs["AX"] & 0xFF)
	var step : int = -1 if cpu_flags["DF"] else 1
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_stosw() -> void:
	cpu_ww("ES", cpu_regs["DI"], cpu_regs["AX"])
	var step : int = -2 if cpu_flags["DF"] else 2
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_lodsb() -> void:
	cpu_set_r8(0, cpu_rb("DS", cpu_regs["SI"]))
	var step : int = -1 if cpu_flags["DF"] else 1
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF

func cpu_lodsw() -> void:
	cpu_regs["AX"] = cpu_rw("DS", cpu_regs["SI"])
	var step : int = -2 if cpu_flags["DF"] else 2
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF

func cpu_movsb() -> void:
	cpu_wb("ES", cpu_regs["DI"], cpu_rb("DS", cpu_regs["SI"]))
	var step : int = -1 if cpu_flags["DF"] else 1
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_movsw() -> void:
	cpu_ww("ES", cpu_regs["DI"], cpu_rw("DS", cpu_regs["SI"]))
	var step : int = -2 if cpu_flags["DF"] else 2
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_scasb() -> void:
	var al  : int = cpu_regs["AX"] & 0xFF
	var mem_val : int = cpu_rb("ES", cpu_regs["DI"])
	var r   : int = al - mem_val
	cpu_upd_flags(r, 8, true, al, mem_val)
	var step : int = -1 if cpu_flags["DF"] else 1
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

# PC10: добавлены CMPSB / CMPSW / SCASW — сравнение и сканирование строк
# словами, отсутствовавшие в базовом наборе PC9 (там был только SCASB).
func cpu_cmpsb() -> void:
	var src_val : int = cpu_rb("DS", cpu_regs["SI"])
	var dst_val : int = cpu_rb("ES", cpu_regs["DI"])
	var r : int = src_val - dst_val
	cpu_upd_flags(r, 8, true, src_val, dst_val)
	var step : int = -1 if cpu_flags["DF"] else 1
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_cmpsw() -> void:
	var src_val : int = cpu_rw("DS", cpu_regs["SI"])
	var dst_val : int = cpu_rw("ES", cpu_regs["DI"])
	var r : int = src_val - dst_val
	cpu_upd_flags(r, 16, true, src_val, dst_val)
	var step : int = -2 if cpu_flags["DF"] else 2
	cpu_regs["SI"] = (cpu_regs["SI"] + step) & 0xFFFF
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

func cpu_scasw() -> void:
	var ax  : int = cpu_regs["AX"]
	var mem_val : int = cpu_rw("ES", cpu_regs["DI"])
	var r   : int = ax - mem_val
	cpu_upd_flags(r, 16, true, ax, mem_val)
	var step : int = -2 if cpu_flags["DF"] else 2
	cpu_regs["DI"] = (cpu_regs["DI"] + step) & 0xFFFF

#  ИСПОЛНЕНИЕ ОПКОДОВ
# ═════════════════════════════════════════════════════════════

func cpu_step() -> void:
	if cpu_halt:
		return
	if cpu_halt_waiting_key:
		if pc11_bios_buffer_is_empty():
			return   # PC11: настоящий busy-wait на клавиатуру — CPU не тратит такты впустую
		cpu_halt_waiting_key = false
	var ip_before : int = cpu_ip
	var op : int = cpu_fetch_b()
	if pc11_logger_verbose:
		pc11_log_instruction(op, ip_before)
	cpu_exec(op)

func cpu_exec(op: int) -> void:
	# ── PC10: новые опкоды (PUSHA/POPA, IMUL imm, INSx/OUTSx,
	#    DAA/AAS, двухбайтовая группа 0x0F) — проверяются первыми,
	#    так как используют коды (0x60,0x61,0x69,0x6B,0x6C-0x6F,
	#    0x27,0x3F,0x0F), свободные в базовой таблице PC9.
	if pc10_try_exec_extra_opcode(op):
		return

	# ── PC11: математический сопроцессор x87 (опкоды 0xD8-0xDF),
	#    диапазон полностью свободен в базовой таблице PC9/PC10.
	if op >= 0xD8 and op <= 0xDF:
		cpu_exec_fpu(op)
		return

	# ── MOV reg, imm ──────────────────────────────────────────
	if op >= 0xB0 and op <= 0xBF:
		var idx : int = op & 7
		if op & 8:
			cpu_regs[REG16[idx]] = cpu_fetch_w()
		else:
			cpu_set_r8(idx, cpu_fetch_b())
		return

	# ── INC/DEC reg16 (0x40-0x4F) ────────────────────────────
	if op >= 0x40 and op <= 0x4F:
		var reg : String = REG16[op & 7]
		var old_cf : int = cpu_flags["CF"]
		var r : int
		if op < 0x48:
			r = (cpu_regs[reg] + 1) & 0xFFFF
			cpu_upd_flags(r, 16, false, cpu_regs[reg], 1)
		else:
			r = (cpu_regs[reg] - 1) & 0xFFFF
			cpu_upd_flags(r, 16, true, cpu_regs[reg], 1)
		cpu_flags["CF"] = old_cf
		cpu_regs[reg] = r
		return

	# ── PUSH reg16 (0x50-0x57) ───────────────────────────────
	if op >= 0x50 and op <= 0x57:
		cpu_push(cpu_regs[REG16[op & 7]])
		return

	# ── POP reg16 (0x58-0x5F) ────────────────────────────────
	if op >= 0x58 and op <= 0x5F:
		cpu_regs[REG16[op & 7]] = cpu_pop()
		return

	# ── XCHG AX, reg (0x91-0x97) ─────────────────────────────
	if op >= 0x91 and op <= 0x97:
		var tmp : int = cpu_regs["AX"]
		cpu_regs["AX"] = cpu_regs[REG16[op & 7]]
		cpu_regs[REG16[op & 7]] = tmp
		return

	# ── Jcc (0x70-0x7F) ──────────────────────────────────────
	if op >= 0x70 and op <= 0x7F:
		_op_jcc(op)
		return

	# ── ALU группы (0x00-0x3F) ───────────────────────────────
	if op <= 0x3F:
		# AAA (0x37) и DAS (0x2F) — особые случаи внутри диапазона ALU
		if op == 0x37:
			_op_aaa()
			return
		if op == 0x2F:
			_op_das()
			return
		var grp  : int  = (op >> 3) & 7
		var word : bool = bool(op & 1)
		var sub_op : int = op & 7
		if sub_op <= 1:          # RM ← REG
			var m : Dictionary = cpu_decode_modrm(word)
			var dst : int = cpu_get_rm(m)
			var src : int = cpu_get_reg(m)
			var res : int = cpu_alu(grp, dst, src, 16 if word else 8)
			if grp != 7: cpu_set_rm(m, res)
		elif sub_op <= 3:        # REG ← RM
			var m : Dictionary = cpu_decode_modrm(word)
			var dst : int = cpu_get_reg(m)
			var src : int = cpu_get_rm(m)
			var res : int = cpu_alu(grp, dst, src, 16 if word else 8)
			if grp != 7: cpu_set_reg(m, res)
		elif sub_op <= 5:        # AX/AL ← imm
			if word:
				var src : int = cpu_fetch_w()
				var res : int = cpu_alu(grp, cpu_regs["AX"], src, 16)
				if grp != 7: cpu_regs["AX"] = res
			else:
				var src : int = cpu_fetch_b()
				var al  : int = cpu_regs["AX"] & 0xFF
				var res : int = cpu_alu(grp, al, src, 8)
				if grp != 7: cpu_set_r8(0, res)
		return

	# ── Большой switch по опкоду ──────────────────────────────
	match op:
		# ─ NOP / HLT ─────────────────────────────────────────
		0x90: pass   # NOP
		0xF4: cpu_halt = true   # HLT

		# ─ Флаги ─────────────────────────────────────────────
		0xF8: cpu_flags["CF"] = 0   # CLC
		0xF9: cpu_flags["CF"] = 1   # STC
		0xF5: cpu_flags["CF"] ^= 1  # CMC
		0xFA: cpu_flags["IF"] = 0   # CLI
		0xFB: cpu_flags["IF"] = 1   # STI
		0xFC: cpu_flags["DF"] = 0   # CLD
		0xFD: cpu_flags["DF"] = 1   # STD

		# ─ MOV r/m, reg ──────────────────────────────────────
		0x88, 0x89:
			var m : Dictionary = cpu_decode_modrm(op == 0x89)
			cpu_set_rm(m, cpu_get_reg(m))
		0x8A, 0x8B:
			var m : Dictionary = cpu_decode_modrm(op == 0x8B)
			cpu_set_reg(m, cpu_get_rm(m))

		# ─ MOV seg ───────────────────────────────────────────
		0x8C:
			var m : Dictionary = cpu_decode_modrm(true)
			var seg_names : Array[String] = ["ES","CS","SS","DS"]
			cpu_set_rm(m, cpu_segs[seg_names[m["reg"] & 3]])
		0x8E:
			var m : Dictionary = cpu_decode_modrm(true)
			var seg_names : Array[String] = ["ES","CS","SS","DS"]
			cpu_segs[seg_names[m["reg"] & 3]] = cpu_get_rm(m)

		# ─ MOV AX,[mem] / [mem],AX ───────────────────────────
		0xA1:
			cpu_regs["AX"] = cpu_rw("DS", cpu_fetch_w())
		0xA3:
			cpu_ww("DS", cpu_fetch_w(), cpu_regs["AX"])

		# ─ MOV r/m, imm ──────────────────────────────────────
		0xC6:
			var m : Dictionary = cpu_decode_modrm(false)
			cpu_set_rm(m, cpu_fetch_b())
		0xC7:
			var m : Dictionary = cpu_decode_modrm(true)
			cpu_set_rm(m, cpu_fetch_w())

		# ─ Группа 1: ALU r/m, imm ────────────────────────────
		0x80, 0x81, 0x83:
			_op_group1(op)

		# ─ Сегментные регистры ───────────────────────────────
		0x06: cpu_push(cpu_segs["ES"])
		0x07: cpu_segs["ES"] = cpu_pop()
		0x0E: cpu_push(cpu_segs["CS"])
		0x16: cpu_push(cpu_segs["SS"])
		0x17: cpu_segs["SS"] = cpu_pop()
		0x1E: cpu_push(cpu_segs["DS"])
		0x1F: cpu_segs["DS"] = cpu_pop()

		# ─ TEST ──────────────────────────────────────────────
		0x84, 0x85:
			var m : Dictionary = cpu_decode_modrm(op == 0x85)
			var res : int = cpu_get_rm(m) & cpu_get_reg(m)
			cpu_upd_flags(res, 16 if op == 0x85 else 8)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		0xA8:
			var res : int = (cpu_regs["AX"] & 0xFF) & cpu_fetch_b()
			cpu_upd_flags(res, 8)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		0xA9:
			var res : int = cpu_regs["AX"] & cpu_fetch_w()
			cpu_upd_flags(res, 16)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0

		# ─ XCHG r/m, reg ─────────────────────────────────────
		0x86, 0x87:
			var m : Dictionary = cpu_decode_modrm(op == 0x87)
			var tmp : int = cpu_get_rm(m)
			cpu_set_rm(m, cpu_get_reg(m))
			cpu_set_reg(m, tmp)

		# ─ LEA ───────────────────────────────────────────────
		0x8D:
			var m : Dictionary = cpu_decode_modrm(true)
			if m["mem"]:
				cpu_regs[REG16[m["reg"]]] = m["off"]

		# ─ LDS / LES ─────────────────────────────────────────
		0xC4, 0xC5:
			var m : Dictionary = cpu_decode_modrm(true)
			if m["mem"]:
				var off_v : int = cpu_rw(m["seg"], m["off"])
				var seg_v : int = cpu_rw(m["seg"], (m["off"] + 2) & 0xFFFF)
				cpu_regs[REG16[m["reg"]]] = off_v
				if op == 0xC5: cpu_segs["DS"] = seg_v
				else:          cpu_segs["ES"] = seg_v

		# ─ BOUND (0x62) ──────────────────────────────────────
		0x62:
			_op_bound()

		# ─ CALL near ─────────────────────────────────────────
		0xE8:
			var off_v : int = cpu_fetch_sw()
			cpu_push(cpu_ip)
			cpu_ip = (cpu_ip + off_v) & 0xFFFF

		# ─ RET ───────────────────────────────────────────────
		0xC3: cpu_ip = cpu_pop()
		0xC2:
			var n : int = cpu_fetch_w()
			cpu_ip = cpu_pop()
			cpu_regs["SP"] = (cpu_regs["SP"] + n) & 0xFFFF
		0xCB:   # RETF
			cpu_ip = cpu_pop()
			cpu_segs["CS"] = cpu_pop()

		# ─ JMP ───────────────────────────────────────────────
		0xEB:
			cpu_ip = (cpu_ip + cpu_fetch_sb()) & 0xFFFF
		0xE9:
			cpu_ip = (cpu_ip + cpu_fetch_sw()) & 0xFFFF
		0xEA:   # JMP far
			var off_v2 : int = cpu_fetch_w()
			var seg_v2 : int = cpu_fetch_w()
			cpu_ip = off_v2
			cpu_segs["CS"] = seg_v2

		# ─ LOOP ──────────────────────────────────────────────
		0xE0, 0xE1, 0xE2:
			_op_loop(op)
		0xE3:
			var off_v3 : int = cpu_fetch_sb()
			if cpu_regs["CX"] == 0:
				cpu_ip = (cpu_ip + off_v3) & 0xFFFF

		# ─ INT ───────────────────────────────────────────────
		0xCC: cpu_handle_interrupt(3)
		0xCD: cpu_handle_interrupt(cpu_fetch_b())
		0xCE:
			if cpu_flags["OF"]:
				cpu_handle_interrupt(4)
		0xCF:   # IRET
			cpu_ip = cpu_pop()
			cpu_segs["CS"] = cpu_pop()
			var fw : int = cpu_pop()
			var iret_names : Array[String] = ["CF","","PF","","AF","","ZF","SF","TF","IF","DF","OF"]
			for i in range(iret_names.size()):
				if iret_names[i] != "":
					cpu_flags[iret_names[i]] = (fw >> i) & 1

		# ─ Сдвиги группы 2 ───────────────────────────────────
		0xD0, 0xD1, 0xD2, 0xD3:
			_op_group2_shift(op)

		# ─ Группа 3 ──────────────────────────────────────────
		0xF6, 0xF7:
			_op_group3(op)

		# ─ Группа 4/5 ────────────────────────────────────────
		0xFE: _op_group4()
		0xFF: _op_group5()

		# ─ Строковые команды ─────────────────────────────────
		0xA4: cpu_movsb()
		0xA5: cpu_movsw()
		0xA6: cpu_cmpsb()   # PC10
		0xA7: cpu_cmpsw()   # PC10
		0xAA: cpu_stosb()
		0xAB: cpu_stosw()
		0xAC: cpu_lodsb()
		0xAD: cpu_lodsw()
		0xAE: cpu_scasb()
		0xAF: cpu_scasw()   # PC10

		# ─ REP / REPNE ───────────────────────────────────────
		0xF2, 0xF3: _op_rep(op)

		# ─ Сегментные префиксы ───────────────────────────────
		0x26: _op_seg_prefix(cpu_segs["ES"])
		0x2E: _op_seg_prefix(cpu_segs["CS"])
		0x36: _op_seg_prefix(cpu_segs["SS"])
		0x3E: _op_seg_prefix(cpu_segs["DS"])

		# ─ IN / OUT ──────────────────────────────────────────
		0xE4: cpu_set_r8(0, _in_port(cpu_fetch_b()))
		0xE5: cpu_regs["AX"] = _in_port(cpu_fetch_b())
		0xEC: cpu_set_r8(0, _in_port(cpu_regs["DX"]))
		0xED: cpu_regs["AX"] = _in_port(cpu_regs["DX"])
		0xE6: _out_port(cpu_fetch_b(), cpu_regs["AX"] & 0xFF)
		0xE7: _out_port(cpu_fetch_b(), cpu_regs["AX"])
		0xEE: _out_port(cpu_regs["DX"], cpu_regs["AX"] & 0xFF)
		0xEF: _out_port(cpu_regs["DX"], cpu_regs["AX"])

		# ─ PUSH immediate (новое: 80186-стиль, добавлено в PC9) ─
		0x6A:   # PUSH imm8 (знаковое расширение до 16 бит)
			var v8 : int = cpu_fetch_sb()
			cpu_push(v8 & 0xFFFF)
		0x68:   # PUSH imm16
			var v16 : int = cpu_fetch_w()
			cpu_push(v16)

		# ─ ENTER / LEAVE (новое: удобно для скомпилированных
		#   программ на Си, добавлено в PC9) ──────────────────
		0xC8:   # ENTER imm16, imm8
			var alloc_size : int = cpu_fetch_w()
			var nesting    : int = cpu_fetch_b() & 0x1F
			cpu_push(cpu_regs["BP"])
			var frame_temp : int = cpu_regs["SP"]
			if nesting > 0:
				for lvl in range(1, nesting):
					cpu_regs["BP"] = (cpu_regs["BP"] - 2) & 0xFFFF
					cpu_push(cpu_rw("SS", cpu_regs["BP"]))
				cpu_push(frame_temp)
			cpu_regs["BP"] = frame_temp
			cpu_regs["SP"] = (cpu_regs["SP"] - alloc_size) & 0xFFFF
		0xC9:   # LEAVE
			cpu_regs["SP"] = cpu_regs["BP"]
			cpu_regs["BP"] = cpu_pop()

		# ─ Прочее ────────────────────────────────────────────
		0xD4: _op_aam()
		0xD5: _op_aad()
		0x98: _op_cbw()
		0x99: _op_cwd()
		0xD7: _op_xlat()
		0x9C: _op_pushf()
		0x9D: _op_popf()
		0x9E: _op_sahf()
		0x9F: _op_lahf()

		_:
			print("[CPU ERR] Неизвестный опкод 0x%02X на CS:IP=%04X:%04X" % [
				op, cpu_segs["CS"], (cpu_ip - 1) & 0xFFFF])
			cpu_handle_interrupt(0x06)   # PC9: недопустимый опкод -> INT 06h

# ── Группа 1 ──────────────────────────────────────────────────
func _op_group1(op: int) -> void:
	var word : bool = (op != 0x80)
	var m    : Dictionary = cpu_decode_modrm(word)
	var imm  : int
	if op == 0x81:
		imm = cpu_fetch_w()
	elif op == 0x83:
		imm = cpu_fetch_sb()
		if imm < 0: imm &= 0xFFFF
	else:
		imm = cpu_fetch_b()
	var dst : int = cpu_get_rm(m)
	var res : int = cpu_alu(m["reg"], dst, imm, 16 if word else 8)
	if m["reg"] != 7:
		cpu_set_rm(m, res)

# ── Группа 2: сдвиги ──────────────────────────────────────────
func _op_group2_shift(op: int) -> void:
	var word  : bool = bool(op & 1)
	var m     : Dictionary = cpu_decode_modrm(word)
	var count : int = 1 if op < 0xD2 else (cpu_regs["CX"] & 0xFF)
	var val   : int = cpu_get_rm(m)
	var ext   : int = m["reg"]
	var size  : int = 16 if word else 8
	var mask  : int = 0xFFFF if word else 0xFF

	for _i in range(count & 0x1F):
		match ext:
			4:   # SHL
				cpu_flags["CF"] = (val >> (size - 1)) & 1
				val = (val << 1) & mask
			5:   # SHR
				cpu_flags["CF"] = val & 1
				val = (val >> 1) & mask
			7:   # SAR
				cpu_flags["CF"] = val & 1
				var sign : int = val & (1 << (size - 1))
				val = ((val >> 1) | sign) & mask
			0:   # ROL
				cpu_flags["CF"] = (val >> (size - 1)) & 1
				val = ((val << 1) | cpu_flags["CF"]) & mask
			1:   # ROR
				cpu_flags["CF"] = val & 1
				val = ((cpu_flags["CF"] << (size - 1)) | (val >> 1)) & mask
			2:   # RCL
				var nc : int = (val >> (size - 1)) & 1
				val = ((val << 1) | cpu_flags["CF"]) & mask
				cpu_flags["CF"] = nc
			3:   # RCR
				var nc2 : int = val & 1
				val = ((cpu_flags["CF"] << (size - 1)) | (val >> 1)) & mask
				cpu_flags["CF"] = nc2

	cpu_upd_flags(val, size)
	cpu_set_rm(m, val)

# ── Группа 3 ──────────────────────────────────────────────────
func _op_group3(op: int) -> void:
	var word : bool = bool(op & 1)
	var m    : Dictionary = cpu_decode_modrm(word)
	var val  : int = cpu_get_rm(m)
	var ext  : int = m["reg"]
	var mask : int = 0xFFFF if word else 0xFF

	match ext:
		0:   # TEST r/m, imm
			var imm : int = cpu_fetch_w() if word else cpu_fetch_b()
			var res : int = val & imm
			cpu_upd_flags(res, 16 if word else 8)
			cpu_flags["CF"] = 0; cpu_flags["OF"] = 0
		2:   # NOT
			cpu_set_rm(m, (~val) & mask)
		3:   # NEG
			var res2 : int = (-val) & mask
			cpu_set_rm(m, res2)
			cpu_upd_flags(res2, 16 if word else 8, true, 0, val)
			cpu_flags["CF"] = 1 if val != 0 else 0
		4:   # MUL
			if word:
				var r : int = cpu_regs["AX"] * val
				cpu_regs["AX"] = r & 0xFFFF
				cpu_regs["DX"] = (r >> 16) & 0xFFFF
				cpu_flags["CF"] = 1 if cpu_regs["DX"] != 0 else 0
				cpu_flags["OF"] = cpu_flags["CF"]
			else:
				var r2 : int = (cpu_regs["AX"] & 0xFF) * val
				cpu_regs["AX"] = r2 & 0xFFFF
				cpu_flags["CF"] = 1 if r2 > 0xFF else 0
				cpu_flags["OF"] = cpu_flags["CF"]
		5:   # IMUL
			if word:
				var a : int = cpu_regs["AX"] if cpu_regs["AX"] < 0x8000 else cpu_regs["AX"] - 0x10000
				var b : int = val if val < 0x8000 else val - 0x10000
				var r3 : int = a * b
				cpu_regs["AX"] = r3 & 0xFFFF
				cpu_regs["DX"] = (r3 >> 16) & 0xFFFF
			else:
				var a8 : int = cpu_regs["AX"] & 0xFF
				if a8 >= 0x80: a8 -= 0x100
				var b8 : int = val if val < 0x80 else val - 0x100
				var r4 : int = a8 * b8
				cpu_regs["AX"] = r4 & 0xFFFF
		6:   # DIV
			if val == 0:
				cpu_handle_interrupt(0x00)   # PC9: деление на ноль -> настоящее INT 00h
				return
			if word:
				var num : int = (cpu_regs["DX"] << 16) | cpu_regs["AX"]
				var q   : int = int(num / val)
				var rmv : int = num % val
				if q > 0xFFFF:
					cpu_handle_interrupt(0x00)
					return
				cpu_regs["AX"] = q & 0xFFFF
				cpu_regs["DX"] = rmv & 0xFFFF
			else:
				var numb : int = cpu_regs["AX"]
				var qb   : int = int(numb / val)
				var rmb  : int = numb % val
				if qb > 0xFF:
					cpu_handle_interrupt(0x00)
					return
				cpu_set_r8(0, qb)
				cpu_set_r8(4, rmb)
		7:   # IDIV
			if val == 0:
				cpu_handle_interrupt(0x00)
				return
			if word:
				var numw : int = (cpu_regs["DX"] << 16) | cpu_regs["AX"]
				if numw >= 0x80000000: numw -= 0x100000000
				var sv  : int = val if val < 0x8000 else val - 0x10000
				if sv == 0:
					cpu_handle_interrupt(0x00)
					return
				var qw  : int = int(numw / sv)
				var rmw : int = numw - qw * sv
				cpu_regs["AX"] = qw & 0xFFFF
				cpu_regs["DX"] = rmw & 0xFFFF

# ── Группа 4 ──────────────────────────────────────────────────
func _op_group4() -> void:
	var m   : Dictionary = cpu_decode_modrm(false)
	var val : int = cpu_get_rm(m)
	var ext : int = m["reg"]
	var old_cf : int = cpu_flags["CF"]
	var r : int = (val + 1) & 0xFF if ext == 0 else (val - 1) & 0xFF
	cpu_upd_flags(r, 8, ext == 1, val, 1)
	cpu_flags["CF"] = old_cf
	cpu_set_rm(m, r)

# ── Группа 5 ──────────────────────────────────────────────────
func _op_group5() -> void:
	var m   : Dictionary = cpu_decode_modrm(true)
	var val : int = cpu_get_rm(m)
	var ext : int = m["reg"]
	var old_cf : int = cpu_flags["CF"]
	match ext:
		0:   # INC
			var r : int = (val + 1) & 0xFFFF
			cpu_upd_flags(r, 16, false, val, 1)
			cpu_flags["CF"] = old_cf
			cpu_set_rm(m, r)
		1:   # DEC
			var r2 : int = (val - 1) & 0xFFFF
			cpu_upd_flags(r2, 16, true, val, 1)
			cpu_flags["CF"] = old_cf
			cpu_set_rm(m, r2)
		2:   # CALL near indirect
			cpu_push(cpu_ip)
			cpu_ip = val
		4:   # JMP near indirect
			cpu_ip = val
		6:   # PUSH r/m
			cpu_push(val)

# ── REP ───────────────────────────────────────────────────────
func _op_rep(op: int) -> void:
	var repne : bool = (op == 0xF2)
	var sc    : int  = cpu_fetch_b()
	while cpu_regs["CX"] > 0:
		match sc:
			0xA4: cpu_movsb()
			0xA5: cpu_movsw()
			0xA6: cpu_cmpsb()   # PC10
			0xA7: cpu_cmpsw()   # PC10
			0xAA: cpu_stosb()
			0xAB: cpu_stosw()
			0xAC: cpu_lodsb()
			0xAD: cpu_lodsw()
			0xAE: cpu_scasb()
			0xAF: cpu_scasw()   # PC10
			_:
				print("[REP] Неизвестная команда 0x%02X" % sc)
				cpu_halt = true
				break
		cpu_regs["CX"] = (cpu_regs["CX"] - 1) & 0xFFFF
		# PC10: условие прерывания REPE/REPNE теперь распространено на
		# все команды сравнения/сканирования (SCASB/SCASW/CMPSB/CMPSW),
		# а не только на SCASB, как было в базовом наборе PC9.
		if sc in [0xAE, 0xAF, 0xA6, 0xA7]:
			if repne and cpu_flags["ZF"] == 1: break
			if not repne and cpu_flags["ZF"] == 0: break

# ── Сегментный префикс ────────────────────────────────────────
func _op_seg_prefix(override_val: int) -> void:
	cpu_seg_override = override_val
	var next_op : int = cpu_fetch_b()
	cpu_exec(next_op)
	cpu_seg_override = -1

# ── Условные переходы ─────────────────────────────────────────
func _op_jcc(op: int) -> void:
	var off_v : int = cpu_fetch_sb()
	var f     : Dictionary = cpu_flags
	var cond  : bool = false
	match op:
		0x70: cond = f["OF"] == 1
		0x71: cond = f["OF"] == 0
		0x72: cond = f["CF"] == 1
		0x73: cond = f["CF"] == 0
		0x74: cond = f["ZF"] == 1
		0x75: cond = f["ZF"] == 0
		0x76: cond = f["CF"] == 1 or f["ZF"] == 1
		0x77: cond = f["CF"] == 0 and f["ZF"] == 0
		0x78: cond = f["SF"] == 1
		0x79: cond = f["SF"] == 0
		0x7A: cond = f["PF"] == 1
		0x7B: cond = f["PF"] == 0
		0x7C: cond = f["SF"] != f["OF"]
		0x7D: cond = f["SF"] == f["OF"]
		0x7E: cond = f["ZF"] == 1 or f["SF"] != f["OF"]
		0x7F: cond = f["ZF"] == 0 and f["SF"] == f["OF"]
	if cond:
		cpu_ip = (cpu_ip + off_v) & 0xFFFF

# ── LOOP ──────────────────────────────────────────────────────
func _op_loop(op: int) -> void:
	var off_v : int = cpu_fetch_sb()
	cpu_regs["CX"] = (cpu_regs["CX"] - 1) & 0xFFFF
	var do_jump : bool = cpu_regs["CX"] != 0
	if op == 0xE1: do_jump = do_jump and cpu_flags["ZF"] == 1
	if op == 0xE0: do_jump = do_jump and cpu_flags["ZF"] == 0
	if do_jump:
		cpu_ip = (cpu_ip + off_v) & 0xFFFF

# ── AAM / AAD / CBW / CWD / XLAT / PUSHF / POPF / SAHF / LAHF ─
func _op_aam() -> void:
	var base : int = cpu_fetch_b()
	if base == 0:
		cpu_handle_interrupt(0x00)
		return
	var al : int = cpu_regs["AX"] & 0xFF
	cpu_regs["AX"] = ((al / base) << 8) | (al % base)
	cpu_upd_flags(cpu_regs["AX"] & 0xFF, 8)

func _op_aad() -> void:
	var base : int = cpu_fetch_b()
	var ah   : int = (cpu_regs["AX"] >> 8) & 0xFF
	var al   : int =  cpu_regs["AX"]       & 0xFF
	var res  : int = (al + ah * base) & 0xFF
	cpu_regs["AX"] = res
	cpu_upd_flags(res, 8)

func _op_cbw() -> void:
	var al : int = cpu_regs["AX"] & 0xFF
	cpu_regs["AX"] = al if al < 0x80 else (al | 0xFF00)

func _op_cwd() -> void:
	cpu_regs["DX"] = 0xFFFF if (cpu_regs["AX"] & 0x8000) != 0 else 0

func _op_xlat() -> void:
	var al  : int = cpu_regs["AX"] & 0xFF
	var bx  : int = cpu_regs["BX"]
	cpu_set_r8(0, cpu_rb("DS", (bx + al) & 0xFFFF))

func _op_pushf() -> void:
	var fw     : int = 0
	var names  : Array[String] = ["CF","","PF","","AF","","ZF","SF","TF","IF","DF","OF"]
	for i in range(names.size()):
		if names[i] != "":
			fw |= cpu_flags.get(names[i], 0) << i
	cpu_push(fw)

func _op_popf() -> void:
	var fw    : int = cpu_pop()
	var names : Array[String] = ["CF","","PF","","AF","","ZF","SF","TF","IF","DF","OF"]
	for i in range(names.size()):
		if names[i] != "":
			cpu_flags[names[i]] = (fw >> i) & 1

func _op_sahf() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	cpu_flags["CF"] = (ah >> 0) & 1
	cpu_flags["PF"] = (ah >> 2) & 1
	cpu_flags["AF"] = (ah >> 4) & 1
	cpu_flags["ZF"] = (ah >> 6) & 1
	cpu_flags["SF"] = (ah >> 7) & 1

func _op_lahf() -> void:
	var ah : int = (cpu_flags["CF"]
		| (cpu_flags["PF"] << 2)
		| (cpu_flags["AF"] << 4)
		| (cpu_flags["ZF"] << 6)
		| (cpu_flags["SF"] << 7)) | 0x02
	cpu_regs["AX"] = (cpu_regs["AX"] & 0x00FF) | (ah << 8)

# ── PC9: AAA / DAS — BCD-коррекция (добавлено по списку задач) ─
func _op_aaa() -> void:
	var al : int = cpu_regs["AX"] & 0xFF
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	if (al & 0x0F) > 9 or cpu_flags["AF"] == 1:
		al = (al + 6) & 0xFF
		ah = (ah + 1) & 0xFF
		cpu_flags["AF"] = 1
		cpu_flags["CF"] = 1
	else:
		cpu_flags["AF"] = 0
		cpu_flags["CF"] = 0
	al = al & 0x0F
	cpu_regs["AX"] = (ah << 8) | al

func _op_das() -> void:
	var al     : int = cpu_regs["AX"] & 0xFF
	var old_cf : int = cpu_flags["CF"]
	var old_al : int = al
	cpu_flags["CF"] = 0
	if (al & 0x0F) > 9 or cpu_flags["AF"] == 1:
		al = (al - 6) & 0xFF
		cpu_flags["AF"] = 1
	else:
		cpu_flags["AF"] = 0
	if old_al > 0x99 or old_cf == 1:
		al = (al - 0x60) & 0xFF
		cpu_flags["CF"] = 1
	cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00) | al
	cpu_upd_flags(al, 8)

# ── PC9: BOUND — проверка границ массива (добавлено по списку) ─
func _op_bound() -> void:
	var m : Dictionary = cpu_decode_modrm(true)
	if not m["mem"]:
		# BOUND с операндом-регистром недопустим — просто игнорируем
		return
	var index_value : int = cpu_regs[REG16[m["reg"]]]
	if index_value & 0x8000: index_value -= 0x10000
	var low_bound  : int = cpu_rw(m["seg"], m["off"])
	var high_bound : int = cpu_rw(m["seg"], (m["off"] + 2) & 0xFFFF)
	if low_bound & 0x8000: low_bound -= 0x10000
	if high_bound & 0x8000: high_bound -= 0x10000
	if index_value < low_bound or index_value > high_bound:
		_ps_print("[CPU EXCEPTION] BOUND: индекс %d вне диапазона [%d, %d]\n" % [index_value, low_bound, high_bound])
		cpu_handle_interrupt(0x05)

# ── Порты (CGA статус-регистр + AdLib OPL2, добавлено в PC9) ──
func _in_port(port: int) -> int:
	if port == 0x60:
		if not pc11_bios_buffer_is_empty():
			return pc11_peek_bios_buffer()["scan"]
		return 0
	if port == 0x3DA:
		var t : int = int(Time.get_ticks_msec() / 10) % 2
		return 0x08 if t == 0 else 0x00
	if port == 0x388:
		return 0x00   # Статус AdLib: 00 = готов принимать команды
	if port >= 0x1F0 and port <= 0x1F7:
		return read_hardware_port_ide(port)   # PC10: ATA/IDE контроллер
	if port == 0x384:
		return tiga_read_vram_at_hstadr() & 0xFF   # PC10: TIGA HSTDATA (lo byte)
	if port >= 0x3F8 and port <= 0x3FF:
		return pc11_uart_port_read(port)   # PC11: COM1 UART 8250/16450
	return 0xFF

func _out_port(port: int, val: int) -> void:
	if port in [0x42, 0x43, 0x61]:
		pass   # PC-speaker (зарезервировано)
	elif port == 0x388:
		adlib_index = val & 0xFF
	elif port == 0x389:
		adlib_write_register(adlib_index, val & 0xFF)
	elif port == 0x20:
		pass   # EOI контроллера прерываний (зарезервировано)
	elif port >= 0x1F0 and port <= 0x1F7:
		write_hardware_port_ide(port, val)   # PC10: ATA/IDE контроллер
	elif port == 0x380:
		tiga_hstadr_reg = (tiga_hstadr_reg & 0xFFFFFF00) | (val & 0xFF)
	elif port == 0x381:
		tiga_hstadr_reg = (tiga_hstadr_reg & 0xFFFF00FF) | ((val & 0xFF) << 8)
	elif port == 0x384:
		tiga_write_vram_at_hstadr(val & 0xFF)
	elif port >= 0x3F8 and port <= 0x3FF:
		pc11_uart_port_write(port, val)   # PC11: COM1 UART 8250/16450

# ══════════════════════════════════════════════════════════════
#  PC11 — ЛОГГЕР "СУРОВОЙ ПРАВДЫ" (диагностика CPU и прерываний)
#  Опциональный режим трассировки: логирует каждый исполненный
#  опкод и каждое прерывание прямо в системную консоль Godot.
#  Включается/выключается через pc11_logger_verbose, по умолчанию
#  выключен, чтобы не замедлять обычную работу эмулятора.
# ══════════════════════════════════════════════════════════════
var pc11_logger_verbose : bool = false

const PC11_INT_NAMES : Dictionary = {
	0x10: "VIDEO_BIOS",
	0x16: "KEYBOARD_BIOS",
	0x1A: "SYSTEM_CLOCK",
	0x1C: "TIMER_TICK_HOOK",
	0x20: "PROGRAM_TERMINATE",
	0x21: "DOS_SERVICES",
	0x33: "MOUSE_API",
	0x67: "EMS_MANAGER",
}

func pc11_log_instruction(op : int, ip_at_fetch : int) -> void:
	print("[PC11 CPU] Opcode: 0x%02X  IP: 0x%04X" % [op, ip_at_fetch])

func pc11_log_interrupt(int_num : int) -> void:
	var name : String = PC11_INT_NAMES.get(int_num, "UNKNOWN_INT")
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	var al : int =  cpu_regs["AX"]       & 0xFF
	print("[PC11 INT] INT %02Xh (%s)  AH=%02X AL=%02X" % [int_num, name, ah, al])

func pc11_set_logger_verbose(enabled : bool) -> void:
	pc11_logger_verbose = enabled
	_ps_print("[PC11 LOGGER] Трассировка %s.\n" % ("включена" if enabled else "выключена"))

# ══════════════════════════════════════════════════════════════
#  БЛОК 3 — ПРЕРЫВАНИЯ
# ══════════════════════════════════════════════════════════════
func cpu_handle_interrupt(num: int) -> void:
	if pc11_logger_verbose:
		pc11_log_interrupt(num)
	match num:
		0x00:
			_ps_print("[CPU EXCEPTION] Деление на ноль в %04X:%04X!\n" % [cpu_segs["CS"], cpu_ip])
			cpu_halt = true
		0x05:
			_ps_print("[CPU EXCEPTION] INT 05h: BOUND — выход индекса за границы массива.\n")
		0x06:
			_ps_print("[CPU EXCEPTION] INT 06h: недопустимый опкод.\n")
			cpu_halt = true
		0x08:   # IRQ0 — системный таймер (вызывается из _tick_timer)
			_tick_timer_raw()
		0x10: _int_10h()
		0x11: _int_11h()
		0x12: _int_12h()
		0x15: _int_15h()
		0x16: _int_16h()
		0x1A: _int_1ah()
		0x1C: pass   # PC11: пользовательский timer-tick hook, по умолчанию пуст
		0x33: _int_33h()   # PC11: мышь PS/2 (Microsoft Mouse API)
		0x14: _int_14h()   # PC11: COM1 UART (последовательный порт)
		0x20: cpu_halt = true
		0x21: _int_21h()
		0x25: _int_25h_absolute_disk_read()
		0x26: _int_26h_absolute_disk_write()
		0x27: _int_27h_tsr()
		0x2F: _int_2fh_multiplex()
		0x30: _int_30h_cpm_anchor()
		0x67: _int_67h_ems()
		0x40: pc10_handle_tiga_interrupt(cpu_regs["AX"])   # PC10: TIGA HLE API
		_:
			_ps_print("[BIOS WARNING] Нереализованное прерывание: INT 0x%02X\n" % num)

# ══════════════════════════════════════════════════════════════
#  PC10 — РАСШИРЕННОЕ СИСТЕМНОЕ ЯДРО DOS (INT 20h — INT 30h)
#  Архитектура: прямой доступ к секторам (LBA/DAP), TSR-менеджер,
#  мультиплекс-API, побитовая эмуляция квирка стека INT 25h/26h.
#  Виртуальные диски эмулятора (образы в ОЗУ хоста, не путать с
#  основной памятью memory эмулируемого 8086).
# ══════════════════════════════════════════════════════════════
var pc10_drive_a_sectors : PackedByteArray
var pc10_drive_c_sectors : PackedByteArray

func pc10_disks_init() -> void:
	pc10_drive_c_sectors = PackedByteArray()
	pc10_drive_c_sectors.resize(10 * 1024 * 1024)
	pc10_drive_c_sectors.fill(0x00)
	pc10_drive_a_sectors = PackedByteArray()
	pc10_drive_a_sectors.resize(1440 * 1024)
	pc10_drive_a_sectors.fill(0xF6)

func _pc10_simulate_int25_26_stack_quirk() -> void:
	var ss : int = cpu_segs["SS"]
	var sp : int = (cpu_regs["SP"] - 2) & 0xFFFF
	cpu_regs["SP"] = sp
	var phys : int = ((ss << 4) + sp) & 0xFFFFF
	var fw : int = 0
	var names : Array[String] = ["CF","","PF","","AF","","ZF","SF","TF","IF","DF","OF"]
	for i in range(names.size()):
		if names[i] != "":
			fw |= cpu_flags.get(names[i], 0) << i
	if phys + 1 < RAM_SIZE:
		memory[phys]     = fw & 0xFF
		memory[phys + 1] = (fw >> 8) & 0xFF

func _int_absolute_disk_io(is_read: bool) -> void:
	var drive        : int = cpu_regs["AX"] & 0xFF   # AL
	var num_sectors  : int = cpu_regs["CX"]
	var start_sector : int = cpu_regs["DX"]
	var ds  : int = cpu_segs["DS"]
	var bx  : int = cpu_regs["BX"]
	var target_mem_addr : int = (ds << 4) + bx

	var selected_drive_bytes : PackedByteArray = pc10_drive_c_sectors if drive >= 2 else pc10_drive_a_sectors

	if num_sectors == 0xFFFF:
		var dap_addr : int = target_mem_addr
		if dap_addr + 11 < RAM_SIZE:
			start_sector = (memory[dap_addr + 4]
				| (memory[dap_addr + 5] << 8)
				| (memory[dap_addr + 6] << 16)
				| (memory[dap_addr + 7] << 24))
			num_sectors = memory[dap_addr + 2] | (memory[dap_addr + 3] << 8)
			var buf_offset  : int = memory[dap_addr + 8] | (memory[dap_addr + 9] << 8)
			var buf_segment : int = memory[dap_addr + 10] | (memory[dap_addr + 11] << 8)
			target_mem_addr = (buf_segment << 4) + buf_offset

	var disk_offset       : int = start_sector * IDE_SECTOR_SIZE
	var total_bytes_to_io : int = num_sectors * IDE_SECTOR_SIZE

	if disk_offset + total_bytes_to_io > selected_drive_bytes.size() or target_mem_addr + total_bytes_to_io > RAM_SIZE:
		cpu_flags["CF"] = 1
		cpu_regs["AX"]  = 0x0201
		_pc10_simulate_int25_26_stack_quirk()
		return

	if is_read:
		for i in range(total_bytes_to_io):
			memory[target_mem_addr + i] = selected_drive_bytes[disk_offset + i]
	else:
		for i in range(total_bytes_to_io):
			selected_drive_bytes[disk_offset + i] = memory[target_mem_addr + i]

	cpu_flags["CF"] = 0
	_pc10_simulate_int25_26_stack_quirk()

func _int_25h_absolute_disk_read() -> void:
	_int_absolute_disk_io(true)

func _int_26h_absolute_disk_write() -> void:
	_int_absolute_disk_io(false)

func _int_27h_tsr() -> void:
	var dx : int = cpu_regs["DX"]
	var cs : int = cpu_segs["CS"]
	var resident_size_paragraphs : int = ((dx + 15) >> 4)
	_ps_print("[DOS TSR] Программа по адресу %04X:0000 осталась резидентной. Размер: %d параграфов.\n" % [cs, resident_size_paragraphs])
	cpu_halt = true

func _int_2fh_multiplex() -> void:
	var ax_val : int = cpu_regs["AX"]
	match ax_val:
		0x1600:
			cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00)   # AL=0: Windows Enhanced Mode не запущен
		0x4300:
			cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00) | 0x80   # HIMEM.SYS обнаружен
		0x4310:
			cpu_segs["ES"] = 0x9000
			cpu_regs["BX"] = 0x1000
		_:
			cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00)

func _int_30h_cpm_anchor() -> void:
	cpu_flags["CF"] = 0
	_ps_print("[DOS CP/M] Вызов заглушки INT 30h для совместимости с ПО конца 70-х.\n")

# ── INT 21h — DOS API ─────────────────────────────────────────
func _int_21h() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	var al : int =  cpu_regs["AX"]       & 0xFF

	match ah:
		0x00: cpu_halt = true
		0x02:
			var ch : String = char(cpu_regs["DX"] & 0xFF)
			_ps_print(ch)
		0x09:
			var addr : int = cpu_get_phys("DS", cpu_regs["DX"])
			var s    : String = ""
			while memory[addr] != 0x24 and addr < RAM_SIZE - 1:
				s += char(memory[addr])
				addr += 1
			_ps_print(s)
		0x2A:
			var dt : Dictionary = Time.get_datetime_dict_from_system()
			cpu_regs["CX"] = dt["year"]
			cpu_regs["DX"] = (dt["month"] << 8) | dt["day"]
		0x2C:
			var dt2 : Dictionary = Time.get_datetime_dict_from_system()
			cpu_regs["CX"] = (dt2["hour"] << 8) | dt2["minute"]
			cpu_regs["DX"] = (dt2["second"] << 8) | 0
		0x30:
			cpu_regs["AX"] = 0x0003  # DOS 3.0
		0x35:
			cpu_regs["BX"] = 0
			cpu_segs["ES"] = 0
		0x3C:
			pc11_create_file()
		0x3D:
			pc11_open_file()
		0x3E:
			pc11_close_file()
		0x3F:
			pc11_read_file()
		0x40:
			var fd4   : int = cpu_regs["BX"]
			var cnt2  : int = cpu_regs["CX"]
			var addr4 : int = cpu_get_phys("DS", cpu_regs["DX"])
			if fd4 in [1, 2]:
				var s2 : String = ""
				for i in range(cnt2):
					if addr4 + i < RAM_SIZE:
						s2 += char(memory[addr4 + i])
				_ps_print(s2)
				cpu_regs["AX"] = cnt2
				cpu_flags["CF"] = 0
			else:
				pc11_write_file()
		0x41:
			pc11_delete_file()
		0x42:
			pc11_seek_file()
		0x43:
			pc11_attrib_file()
		0x45:
			pc11_dup_handle()
		0x46:
			pc11_force_dup_handle()
		0x56:
			pc11_rename_file()
		0x57:
			pc11_filetime()
		0x5A:
			pc11_create_temp()
		0x48:
			cpu_regs["AX"] = 0x2000
			cpu_flags["CF"] = 0
		0x49: cpu_flags["CF"] = 0
		0x4A: cpu_flags["CF"] = 0
		0x4C:
			_ps_print("\n[DOS] Программа завершена. Код: %d\n" % al)
			cpu_halt = true
		0xEE:   # PC10: кастомная функция — авторизация аккаунта через DOS API
			pc10_int21h_ee_login()
		_: pass

# ══════════════════════════════════════════════════════════════
#  PC11 — РАСШИРЕННАЯ ФАЙЛОВАЯ СИСТЕМА (надстройка над DOS INT 21h)
#  Использует общий cpu_file_handles (fd -> FileAccess) из базы PC10,
#  добавляя поверх него: пул хэндлов с переиспользованием, хранение
#  путей/режимов на каждый fd, CREATE/DELETE/SEEK/ATTRIB/DUP/RENAME/
#  FILETIME/CREATETEMP — то, чего не было в базовом наборе PC10.
# ══════════════════════════════════════════════════════════════

func pc11_fs_init() -> void:
	for h in cpu_file_handles.keys():
		cpu_file_handles[h].close()
	cpu_file_handles.clear()
	pc11_file_paths.clear()
	pc11_file_modes.clear()
	pc11_free_handles.clear()
	pc11_next_handle = PC11_RESERVED_HANDLES
	pc11_last_error = 0
	var dir : DirAccess = DirAccess.open("user://")
	if dir and not dir.dir_exists(PC11_BASE_PATH):
		dir.make_dir_recursive(PC11_BASE_PATH)

func pc11_alloc_handle() -> int:
	if pc11_free_handles.size() > 0:
		return pc11_free_handles.pop_back()
	var candidate : int = pc11_next_handle
	if cpu_file_handles.size() > 0:
		candidate = max(candidate, cpu_file_handles.keys().max() + 1)
	if candidate >= PC11_MAX_HANDLES:
		return -1
	pc11_next_handle = candidate + 1
	return candidate

func pc11_release_handle(handle : int) -> void:
	if not pc11_free_handles.has(handle):
		pc11_free_handles.append(handle)

func pc11_resolve_path(filename : String) -> String:
	var clean : String = filename.replace("\\", "/")
	if clean.begins_with("/"):
		clean = clean.substr(1, clean.length() - 1)
	return PC11_BASE_PATH + clean

func pc11_read_asciiz(addr_start : int) -> String:
	var addr : int = addr_start
	var s : String = ""
	while addr < RAM_SIZE - 1 and memory[addr] != 0:
		s += char(memory[addr])
		addr += 1
	return s

func pc11_create_file() -> void:
	var addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	var filename : String = pc11_read_asciiz(addr)
	var path : String = pc11_resolve_path(filename)
	var dir : DirAccess = DirAccess.open("user://")
	if dir:
		dir.make_dir_recursive(path.get_base_dir())
	var f : FileAccess = FileAccess.open(path, FileAccess.WRITE)
	if f:
		var handle : int = pc11_alloc_handle()
		if handle < 0:
			f.close()
			cpu_regs["AX"] = 4
			cpu_flags["CF"] = 1
			pc11_last_error = 4
			return
		cpu_file_handles[handle] = f
		pc11_file_paths[handle] = path
		pc11_file_modes[handle] = FileAccess.WRITE
		cpu_regs["AX"] = handle
		cpu_flags["CF"] = 0
		pc11_last_error = 0
	else:
		cpu_regs["AX"] = 5
		cpu_flags["CF"] = 1
		pc11_last_error = 5

func pc11_open_file() -> void:
	var addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	var filename : String = pc11_read_asciiz(addr)
	var path : String = pc11_resolve_path(filename)
	var access_mode : int = cpu_regs["AX"] & 0x07
	if not FileAccess.file_exists(path):
		cpu_regs["AX"] = 2
		cpu_flags["CF"] = 1
		pc11_last_error = 2
		return
	var mode : int = FileAccess.READ
	match access_mode:
		0: mode = FileAccess.READ
		1: mode = FileAccess.WRITE_READ
		2: mode = FileAccess.READ_WRITE
		_: mode = FileAccess.READ
	var f : FileAccess = FileAccess.open(path, mode)
	if f:
		var handle : int = pc11_alloc_handle()
		if handle < 0:
			f.close()
			cpu_regs["AX"] = 4
			cpu_flags["CF"] = 1
			pc11_last_error = 4
			return
		cpu_file_handles[handle] = f
		pc11_file_paths[handle] = path
		pc11_file_modes[handle] = mode
		cpu_regs["AX"] = handle
		cpu_flags["CF"] = 0
		pc11_last_error = 0
	else:
		cpu_regs["AX"] = 5
		cpu_flags["CF"] = 1
		pc11_last_error = 5

func pc11_close_file() -> void:
	var handle : int = cpu_regs["BX"]
	if cpu_file_handles.has(handle):
		cpu_file_handles[handle].close()
		cpu_file_handles.erase(handle)
		pc11_file_paths.erase(handle)
		pc11_file_modes.erase(handle)
		pc11_release_handle(handle)
		cpu_flags["CF"] = 0
		pc11_last_error = 0
	else:
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6

func pc11_read_file() -> void:
	var handle : int = cpu_regs["BX"]
	var count : int = cpu_regs["CX"]
	var buffer_addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	if not cpu_file_handles.has(handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	if count == 0:
		cpu_regs["AX"] = 0
		cpu_flags["CF"] = 0
		return
	var f : FileAccess = cpu_file_handles[handle]
	var data : PackedByteArray = f.get_buffer(count)
	var written : int = 0
	for i in range(data.size()):
		var dest : int = buffer_addr + i
		if dest < RAM_SIZE:
			memory[dest] = data[i]
			written += 1
	cpu_regs["AX"] = written
	cpu_flags["CF"] = 0
	pc11_last_error = 0

func pc11_write_file() -> void:
	var handle : int = cpu_regs["BX"]
	var count : int = cpu_regs["CX"]
	var buffer_addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	if not cpu_file_handles.has(handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	var f : FileAccess = cpu_file_handles[handle]
	var buffer : PackedByteArray = PackedByteArray()
	for i in range(count):
		var src : int = buffer_addr + i
		if src < RAM_SIZE:
			buffer.append(memory[src])
		else:
			buffer.append(0)
	f.store_buffer(buffer)
	cpu_regs["AX"] = buffer.size()
	cpu_flags["CF"] = 0
	pc11_last_error = 0

func pc11_delete_file() -> void:
	var addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	var filename : String = pc11_read_asciiz(addr)
	var path : String = pc11_resolve_path(filename)
	var dir : DirAccess = DirAccess.open(path.get_base_dir())
	if dir and FileAccess.file_exists(path):
		var err : int = dir.remove(path.get_file())
		if err == OK:
			cpu_flags["CF"] = 0
			pc11_last_error = 0
		else:
			cpu_regs["AX"] = 5
			cpu_flags["CF"] = 1
			pc11_last_error = 5
	else:
		cpu_regs["AX"] = 2
		cpu_flags["CF"] = 1
		pc11_last_error = 2

func pc11_seek_file() -> void:
	var handle : int = cpu_regs["BX"]
	var origin : int = cpu_regs["AX"] & 0xFF
	var offset : int = (cpu_regs["CX"] << 16) | cpu_regs["DX"]
	if not cpu_file_handles.has(handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	var f : FileAccess = cpu_file_handles[handle]
	var new_pos : int = 0
	match origin:
		PC11_SEEK_SET:
			new_pos = offset
		PC11_SEEK_CUR:
			new_pos = f.get_position() + offset
		PC11_SEEK_END:
			new_pos = f.get_length() + offset
		_:
			new_pos = offset
	if new_pos < 0:
		new_pos = 0
	f.seek(new_pos)
	cpu_regs["DX"] = (new_pos >> 16) & 0xFFFF
	cpu_regs["AX"] = new_pos & 0xFFFF
	cpu_flags["CF"] = 0
	pc11_last_error = 0

func pc11_attrib_file() -> void:
	var addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	var filename : String = pc11_read_asciiz(addr)
	var path : String = pc11_resolve_path(filename)
	var sub_function : int = cpu_regs["AX"] & 0xFF
	if not FileAccess.file_exists(path):
		cpu_regs["AX"] = 2
		cpu_flags["CF"] = 1
		pc11_last_error = 2
		return
	match sub_function:
		0:
			cpu_regs["CX"] = 0x20
			cpu_flags["CF"] = 0
		1:
			cpu_flags["CF"] = 0
		_:
			cpu_regs["AX"] = 1
			cpu_flags["CF"] = 1

func pc11_dup_handle() -> void:
	var handle : int = cpu_regs["BX"]
	if not cpu_file_handles.has(handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	var new_handle : int = pc11_alloc_handle()
	if new_handle < 0:
		cpu_regs["AX"] = 4
		cpu_flags["CF"] = 1
		pc11_last_error = 4
		return
	var path : String = pc11_file_paths.get(handle, "")
	var mode : int = pc11_file_modes.get(handle, FileAccess.READ)
	var dup_f : FileAccess = FileAccess.open(path, mode)
	if dup_f:
		dup_f.seek(cpu_file_handles[handle].get_position())
		cpu_file_handles[new_handle] = dup_f
		pc11_file_paths[new_handle] = path
		pc11_file_modes[new_handle] = mode
		cpu_regs["AX"] = new_handle
		cpu_flags["CF"] = 0
		pc11_last_error = 0
	else:
		pc11_release_handle(new_handle)
		cpu_regs["AX"] = 4
		cpu_flags["CF"] = 1
		pc11_last_error = 4

func pc11_force_dup_handle() -> void:
	var src_handle : int = cpu_regs["BX"]
	var target_handle : int = cpu_regs["CX"]
	if not cpu_file_handles.has(src_handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	if cpu_file_handles.has(target_handle):
		cpu_file_handles[target_handle].close()
		cpu_file_handles.erase(target_handle)
	var path : String = pc11_file_paths.get(src_handle, "")
	var mode : int = pc11_file_modes.get(src_handle, FileAccess.READ)
	var dup_f : FileAccess = FileAccess.open(path, mode)
	if dup_f:
		dup_f.seek(cpu_file_handles[src_handle].get_position())
		cpu_file_handles[target_handle] = dup_f
		pc11_file_paths[target_handle] = path
		pc11_file_modes[target_handle] = mode
		cpu_flags["CF"] = 0
		pc11_last_error = 0
	else:
		cpu_regs["AX"] = 4
		cpu_flags["CF"] = 1
		pc11_last_error = 4

func pc11_rename_file() -> void:
	var addr_old : int = cpu_get_phys("DS", cpu_regs["DX"])
	var old_name : String = pc11_read_asciiz(addr_old)
	var old_path : String = pc11_resolve_path(old_name)
	var addr_new : int = cpu_get_phys("DS", cpu_regs["BX"])
	var new_name : String = pc11_read_asciiz(addr_new)
	var new_path : String = pc11_resolve_path(new_name)
	var dir : DirAccess = DirAccess.open(old_path.get_base_dir())
	if dir and FileAccess.file_exists(old_path):
		var err : int = dir.rename(old_path, new_path)
		if err == OK:
			cpu_flags["CF"] = 0
			pc11_last_error = 0
		else:
			cpu_regs["AX"] = 5
			cpu_flags["CF"] = 1
			pc11_last_error = 5
	else:
		cpu_regs["AX"] = 2
		cpu_flags["CF"] = 1
		pc11_last_error = 2

func pc11_filetime() -> void:
	var handle : int = cpu_regs["BX"]
	if not cpu_file_handles.has(handle):
		cpu_regs["AX"] = 6
		cpu_flags["CF"] = 1
		pc11_last_error = 6
		return
	var dt : Dictionary = Time.get_datetime_dict_from_system()
	cpu_regs["DX"] = ((dt["year"] - 1980) << 9) | (dt["month"] << 5) | dt["day"]
	cpu_regs["CX"] = (dt["hour"] << 11) | (dt["minute"] << 5) | int(dt["second"] / 2)
	cpu_flags["CF"] = 0
	pc11_last_error = 0

func pc11_create_temp() -> void:
	var prefix_addr : int = cpu_get_phys("DS", cpu_regs["DX"])
	var prefix : String = pc11_read_asciiz(prefix_addr)
	var rng : RandomNumberGenerator = RandomNumberGenerator.new()
	rng.randomize()
	var temp_name : String = prefix + str(rng.randi_range(1000, 9999)) + ".tmp"
	var path : String = pc11_resolve_path(temp_name)
	var dir : DirAccess = DirAccess.open("user://")
	if dir:
		dir.make_dir_recursive(path.get_base_dir())
	var f : FileAccess = FileAccess.open(path, FileAccess.WRITE_READ)
	if f:
		var handle : int = pc11_alloc_handle()
		if handle < 0:
			f.close()
			cpu_regs["AX"] = 4
			cpu_flags["CF"] = 1
			pc11_last_error = 4
			return
		cpu_file_handles[handle] = f
		pc11_file_paths[handle] = path
		pc11_file_modes[handle] = FileAccess.WRITE_READ
		cpu_regs["AX"] = handle
		cpu_flags["CF"] = 0
		pc11_last_error = 0
		var name_bytes : PackedByteArray = temp_name.to_ascii_buffer()
		for i in range(name_bytes.size()):
			var dest : int = prefix_addr + i
			if dest < RAM_SIZE:
				memory[dest] = name_bytes[i]
		var term : int = prefix_addr + name_bytes.size()
		if term < RAM_SIZE:
			memory[term] = 0
	else:
		cpu_regs["AX"] = 5
		cpu_flags["CF"] = 1
		pc11_last_error = 5

func pc11_close_all_files() -> void:
	for handle in cpu_file_handles.keys():
		cpu_file_handles[handle].close()
	cpu_file_handles.clear()
	pc11_file_paths.clear()
	pc11_file_modes.clear()
	pc11_free_handles.clear()
	pc11_next_handle = PC11_RESERVED_HANDLES

func pc11_get_open_handle_count() -> int:
	return cpu_file_handles.size()

func pc11_handle_is_open(handle : int) -> bool:
	return cpu_file_handles.has(handle)

func pc11_get_file_size(handle : int) -> int:
	if cpu_file_handles.has(handle):
		return cpu_file_handles[handle].get_length()
	return -1

func pc11_get_last_error() -> int:
	return pc11_last_error


# ── INT 10h — Видео BIOS ──────────────────────────────────────
func _int_10h() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	var al : int =  cpu_regs["AX"]       & 0xFF
	var bh : int = (cpu_regs["BX"] >> 8) & 0xFF
	match ah:
		0x00:   # PC11: установка видеорежима (VideoController, AH=00h)
			if al == 0x03:
				pc11_clear_screen(0x07)
				pc11_set_cursor_pos(0, 0, 0)
			else:
				_ps_print("[INT 10h] Неподдерживаемый видеорежим: %02X\n" % al)
		0x02:   # PC11: установка позиции курсора (AH=02h)
			var row : int = (cpu_regs["DX"] >> 8) & 0xFF
			var col : int =  cpu_regs["DX"]       & 0xFF
			pc11_set_cursor_pos(col, row, bh)
		0x03:   # PC11: чтение позиции курсора (AH=03h)
			var pos : Dictionary = pc11_get_cursor_pos(bh)
			cpu_regs["DX"] = (pos["y"] << 8) | pos["x"]
		0x06:   # PC11: прокрутка/очистка окна (AH=06h)
			var lines : int = al
			var color_attr : int = bh
			if lines == 0:
				pc11_clear_screen(color_attr)
			else:
				pc11_scroll_up(lines, color_attr)
		0x0E: pc11_print_char_teletype(al)
		_: pass

# ──────────────────────────────────────────────────────────────
#  PC11: ЧЕСТНАЯ ТЕКСТОВАЯ VRAM (0xB8000) + КУРСОР В BDA
#  В отличие от базового PC10 (который просто слал символы в
#  PowerShell), здесь AH=0Eh реально пишет байты символа/атрибута
#  в видеопамять, как на настоящем железе — VGATextDisplay сможет
#  прочитать их оттуда для отрисовки.
# ──────────────────────────────────────────────────────────────
const PC11_BDA_CURSOR_POS : int = 0x0450   # позиции курсора для 8 страниц (2 байта/страница)
const PC11_BDA_VIDEO_PAGE : int = 0x0462

func pc11_set_cursor_pos(col : int, row : int, page : int) -> void:
	var addr : int = PC11_BDA_CURSOR_POS + (page * 2)
	memory[addr]     = col & 0xFF
	memory[addr + 1] = row & 0xFF

func pc11_get_cursor_pos(page : int) -> Dictionary:
	var addr : int = PC11_BDA_CURSOR_POS + (page * 2)
	return {"x": memory[addr], "y": memory[addr + 1]}

func pc11_write_vram_char(col : int, row : int, ascii : int, attr : int) -> void:
	if col < 0 or col >= VGA_TEXT_COLS or row < 0 or row >= VGA_TEXT_ROWS:
		return
	var offset : int = VGA_TEXT_VRAM + (row * VGA_TEXT_COLS + col) * 2
	memory[offset]     = ascii & 0xFF
	memory[offset + 1] = attr  & 0xFF

func pc11_clear_screen(attr : int) -> void:
	for i in range(VGA_TEXT_COLS * VGA_TEXT_ROWS):
		var offset : int = VGA_TEXT_VRAM + (i * 2)
		memory[offset]     = 32
		memory[offset + 1] = attr & 0xFF

func pc11_scroll_up(lines : int, attr : int) -> void:
	var bytes_to_scroll : int = (VGA_TEXT_ROWS - lines) * VGA_TEXT_COLS * 2
	var dest_start : int = VGA_TEXT_VRAM
	var src_start  : int = VGA_TEXT_VRAM + (lines * VGA_TEXT_COLS * 2)
	for i in range(bytes_to_scroll):
		memory[dest_start + i] = memory[src_start + i]
	var clear_start : int = VGA_TEXT_VRAM + bytes_to_scroll
	var bytes_to_clear : int = lines * VGA_TEXT_COLS * 2
	var i2 : int = 0
	while i2 < bytes_to_clear:
		memory[clear_start + i2]     = 32
		memory[clear_start + i2 + 1] = attr & 0xFF
		i2 += 2

func pc11_print_char_teletype(ascii : int) -> void:
	var page : int = memory[PC11_BDA_VIDEO_PAGE]
	var pos : Dictionary = pc11_get_cursor_pos(page)
	var col : int = pos["x"]
	var row : int = pos["y"]
	match ascii:
		13:
			col = 0
		10:
			row += 1
		8:
			if col > 0:
				col -= 1
				pc11_write_vram_char(col, row, 0, 0x07)
		_:
			pc11_write_vram_char(col, row, ascii, 0x07)
			col += 1
	if col >= VGA_TEXT_COLS:
		col = 0
		row += 1
	if row >= VGA_TEXT_ROWS:
		pc11_scroll_up(1, 0x07)
		row = VGA_TEXT_ROWS - 1
	pc11_set_cursor_pos(col, row, page)
	_ps_print(char(ascii))   # дублируем в PowerShell-консоль, как было в базовом PC10

# ── INT 11h — Опрос конфигурации оборудования (новое в PC9) ──
func _int_11h() -> void:
	# Биты результата: 0=1 (есть дисковод), 4-5=10 (видеорежим CGA 80x25),
	# 6-7=00 (количество дисководов минус 1)
	cpu_regs["AX"] = 0x0021

# ── INT 12h — Размер базовой памяти (новое в PC9) ─────────────
func _int_12h() -> void:
	cpu_regs["AX"] = 640   # Сообщаем программе, что доступно 640 КБ ОЗУ

# ── INT 15h — Системные сервисы: задержки/джойстик (новое) ───
func _int_15h() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	match ah:
		0x86:   # Задержка в микросекундах (CX:DX)
			var cx : int = cpu_regs["CX"]
			var dx : int = cpu_regs["DX"]
			var microseconds : int = (cx << 16) | dx
			OS.delay_usec(mini(microseconds, 200000))   # ограничение, чтобы не подвесить эмулятор
			cpu_flags["CF"] = 0
		0x84:   # Поддержка джойстика
			var sub_fn : int = cpu_regs["DX"]
			if sub_fn == 0:
				cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00) | 0xF0   # кнопки не нажаты
				cpu_flags["CF"] = 0
			else:
				cpu_regs["AX"] = 0
				cpu_regs["BX"] = 0
				cpu_regs["CX"] = 0
				cpu_regs["DX"] = 0
				cpu_flags["CF"] = 1   # джойстик не подключён
		0x88:   # Память выше 1 МБ
			cpu_regs["AX"] = 0
			cpu_flags["CF"] = 0
		0x87:   # PC10: Move Extended Memory Block (через GDT, 3 МБ шина)
			pc10_handle_int15_move_ext_memory()
		_:
			cpu_flags["CF"] = 0

# ──────────────────────────────────────────────────────────────
#  PC11: НАСТОЯЩИЙ BDA-КОЛЬЦЕВОЙ БУФЕР КЛАВИАТУРЫ (как на железе)
#  В отличие от базового PC10, где буфер хранился в обычном
#  GDScript-массиве cpu_keyboard_buf, здесь скан-код и ASCII-код
#  реально лежат в памяти эмулятора по адресам 0x041E-0x043E,
#  а указатели "голова"/"хвост" — в BDA по 0x041A/0x041C, точно
#  как описано в IBM PC BIOS Technical Reference.
# ──────────────────────────────────────────────────────────────
const PC11_SEG_BDA       : int = 0x0400
const PC11_BDA_KBD_HEAD  : int = 0x041A
const PC11_BDA_KBD_TAIL  : int = 0x041C
const PC11_BDA_BUF_START : int = 0x001E
const PC11_BDA_BUF_END   : int = 0x003E

func pc11_bda_kbd_init() -> void:
	pc11_write_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD, PC11_BDA_BUF_START)
	pc11_write_word_at(PC11_SEG_BDA + PC11_BDA_KBD_TAIL, PC11_BDA_BUF_START)
	memory[BDA_KBD_FLAG1] = 0
	memory[BDA_KBD_FLAG2] = 0

func pc11_write_word_at(addr : int, val : int) -> void:
	memory[addr]     = val & 0xFF
	memory[addr + 1] = (val >> 8) & 0xFF

func pc11_read_word_at(addr : int) -> int:
	return memory[addr] | (memory[addr + 1] << 8)

func pc11_push_to_bios_buffer(scan : int, ascii : int) -> void:
	var head : int = pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD)
	var tail : int = pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_TAIL)
	var next_tail : int = tail + 2
	if next_tail >= PC11_BDA_BUF_END:
		next_tail = PC11_BDA_BUF_START
	if next_tail == head:
		return   # буфер переполнен (на реальном железе тут пищит спикер)
	var abs_addr : int = PC11_SEG_BDA + tail
	memory[abs_addr]     = ascii & 0xFF
	memory[abs_addr + 1] = scan  & 0xFF
	pc11_write_word_at(PC11_SEG_BDA + PC11_BDA_KBD_TAIL, next_tail)
	if cpu_halt_waiting_key:
		cpu_halt_waiting_key = false   # PC11: пробуждаем CPU, если он стоял в HLT/ожидании клавиши

func pc11_pop_from_bios_buffer() -> Dictionary:
	var head : int = pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD)
	var abs_addr : int = PC11_SEG_BDA + head
	var ascii : int = memory[abs_addr]
	var scan  : int = memory[abs_addr + 1]
	var next_head : int = head + 2
	if next_head >= PC11_BDA_BUF_END:
		next_head = PC11_BDA_BUF_START
	pc11_write_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD, next_head)
	return {"scan": scan, "ascii": ascii}

func pc11_peek_bios_buffer() -> Dictionary:
	var head : int = pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD)
	var abs_addr : int = PC11_SEG_BDA + head
	return {"ascii": memory[abs_addr], "scan": memory[abs_addr + 1]}

func pc11_bios_buffer_is_empty() -> bool:
	return pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_HEAD) == pc11_read_word_at(PC11_SEG_BDA + PC11_BDA_KBD_TAIL)

# ── INT 16h — Клавиатура BIOS ────────────────────────────────
func _int_16h() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	match ah:
		0x00, 0x10:
			if pc11_bios_buffer_is_empty():
				cpu_halt_waiting_key = true   # PC11: настоящий флаг ожидания клавиши (как HLT)
				cpu_ip = (cpu_ip - 2) & 0xFFFF  # откатываем IP, чтобы INT 16h выполнился снова
			else:
				var key : Dictionary = pc11_pop_from_bios_buffer()
				cpu_regs["AX"] = (key["scan"] << 8) | key["ascii"]
		0x01, 0x11:
			if pc11_bios_buffer_is_empty():
				cpu_flags["ZF"] = 1
			else:
				var key2 : Dictionary = pc11_peek_bios_buffer()
				cpu_regs["AX"] = (key2["scan"] << 8) | key2["ascii"]
				cpu_flags["ZF"] = 0
		0x02:   # PC11: статус Shift/Ctrl/Alt/Caps (AdvancedKeyboard, AH=02h)
			cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00) | memory[BDA_KBD_FLAG1]
		0x12:   # PC11: расширенный статус (AdvancedKeyboard, AH=12h)
			cpu_regs["AX"] = (memory[BDA_KBD_FLAG2] << 8) | memory[BDA_KBD_FLAG1]
		_: pass

# ── INT 1Ah — Системные часы (новое в PC9) ────────────────────
func _int_1ah() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	match ah:
		0x00:   # Чтение тиков таймера — используется играми для srand()
			var t : int = (memory[TIMER_ADDR]
				| (memory[TIMER_ADDR+1] << 8)
				| (memory[TIMER_ADDR+2] << 16)
				| (memory[TIMER_ADDR+3] << 24))
			cpu_regs["CX"] = (t >> 16) & 0xFFFF
			cpu_regs["DX"] = t & 0xFFFF
			cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00)   # AL=0, полночь не наступала
		0x01:   # Установка тиков таймера
			var t2 : int = (cpu_regs["CX"] << 16) | cpu_regs["DX"]
			memory[TIMER_ADDR]   = t2 & 0xFF
			memory[TIMER_ADDR+1] = (t2 >> 8) & 0xFF
			memory[TIMER_ADDR+2] = (t2 >> 16) & 0xFF
			memory[TIMER_ADDR+3] = (t2 >> 24) & 0xFF
		_: pass

# ══════════════════════════════════════════════════════════════
#  БЛОК 4 — ЗАГРУЗЧИК .COM и .EXE
# ══════════════════════════════════════════════════════════════

func loader_create_psp(segment: int, cmd_line: String = "") -> void:
	var base : int = segment << 4
	for i in range(256):
		memory[base + i] = 0
	memory[base]     = 0xCD
	memory[base + 1] = 0x20
	memory[base + 2] = 0x00
	memory[base + 3] = 0xA0
	var cmd : PackedByteArray = cmd_line.to_ascii_buffer()
	var cmd_len : int = mini(cmd.size(), 126)
	memory[base + 0x80] = cmd_len
	for i in range(cmd_len):
		memory[base + 0x81 + i] = cmd[i]
	memory[base + 0x81 + cmd_len] = 0x0D

func loader_load_com(path: String) -> bool:
	var f : FileAccess = FileAccess.open(path, FileAccess.READ)
	if not f:
		_ps_print("[LOADER] Ошибка чтения: " + path + "\n")
		return false
	var data : PackedByteArray = f.get_buffer(f.get_length())
	f.close()

	var seg  : int = LOAD_SEGMENT
	loader_create_psp(seg, path.get_file())

	var base : int = (seg << 4) + PSP_SIZE
	for i in range(data.size()):
		if base + i < RAM_SIZE:
			memory[base + i] = data[i]

	cpu_segs["CS"] = seg; cpu_segs["DS"] = seg
	cpu_segs["ES"] = seg; cpu_segs["SS"] = seg
	cpu_ip         = 0x0100
	cpu_regs["SP"] = 0xFFFE
	cpu_halt       = false
	_ps_print("[LOADER] .COM загружен: %s (%d байт)\n" % [path, data.size()])
	return true

func loader_load_exe(path: String) -> bool:
	var f : FileAccess = FileAccess.open(path, FileAccess.READ)
	if not f:
		_ps_print("[LOADER] Ошибка чтения: " + path + "\n")
		return false
	var raw : PackedByteArray = f.get_buffer(f.get_length())
	f.close()

	if raw.size() < 28:
		return loader_load_com(path)

	var sig : String = char(raw[0]) + char(raw[1])
	if sig != "MZ" and sig != "ZM":
		_ps_print("[LOADER] Не MZ — загружаем как .COM\n")
		return loader_load_com(path)

	# Разбираем заголовок MZ
	var last_page  : int = (raw[3] << 8) | raw[2]
	var pages      : int = (raw[5] << 8) | raw[4]
	var reloc_cnt  : int = (raw[7] << 8) | raw[6]
	var hdr_para   : int = (raw[9] << 8) | raw[8]
	var ss_rel     : int = (raw[15] << 8) | raw[14]
	var sp_init    : int = (raw[17] << 8) | raw[16]
	var ip_init    : int = (raw[21] << 8) | raw[20]
	var cs_rel     : int = (raw[23] << 8) | raw[22]
	var reloc_off  : int = (raw[25] << 8) | raw[24]

	var hdr_size   : int = hdr_para * 16
	var seg_start  : int = LOAD_SEGMENT
	var phys_start : int = seg_start << 4

	var code : PackedByteArray = raw.slice(hdr_size)
	for i in range(code.size()):
		if phys_start + i < RAM_SIZE:
			memory[phys_start + i] = code[i]

	# Релокация
	var ptr : int = reloc_off
	for _ri in range(reloc_cnt):
		var r_off : int = (raw[ptr + 1] << 8) | raw[ptr]
		var r_seg : int = (raw[ptr + 3] << 8) | raw[ptr + 2]
		ptr += 4
		var fix_addr : int = ((seg_start + r_seg) << 4) + r_off
		if fix_addr + 1 < RAM_SIZE:
			var old_val : int = memory[fix_addr] | (memory[fix_addr + 1] << 8)
			var new_val : int = (old_val + seg_start) & 0xFFFF
			memory[fix_addr]     = new_val & 0xFF
			memory[fix_addr + 1] = (new_val >> 8) & 0xFF

	cpu_segs["CS"] = (seg_start + cs_rel) & 0xFFFF
	cpu_ip         = ip_init
	cpu_segs["SS"] = (seg_start + ss_rel) & 0xFFFF
	cpu_regs["SP"] = sp_init
	cpu_segs["DS"] = seg_start
	cpu_segs["ES"] = seg_start
	cpu_halt       = false
	_ps_print("[LOADER] .EXE загружен: %s (%d байт кода)\n" % [path, code.size()])
	return true

func loader_open_file_dialog() -> void:
	# Открываем нативный проводник Windows через PowerShell + OpenFileDialog
	var tmp_path : String = OS.get_temp_dir().path_join("pc9_selected.txt")

	var tmp_esc  : String = tmp_path.replace("\\", "\\\\")
	var ps_script : String = (
		"Add-Type -AssemblyName System.Windows.Forms;"
		+ "$dlg = New-Object System.Windows.Forms.OpenFileDialog;"
		+ "$dlg.Title = 'PC9: Выберите программу (.COM или .EXE)';"
		+ "$dlg.Filter = 'DOS программы (*.com;*.exe)|*.com;*.exe|Все файлы (*.*)|*.*';"
		+ "$dlg.FilterIndex = 1;"
		+ "if ($dlg.ShowDialog() -eq 'OK') { $dlg.FileName | Out-File -Encoding UTF8 -NoNewline -FilePath '"
		+ tmp_esc + "' }"
		+ " else { Set-Content -Path '" + tmp_esc + "' -Value ''}"
	)

	if FileAccess.file_exists(tmp_path):
		DirAccess.remove_absolute(tmp_path)

	var exit_code : int = OS.execute("powershell.exe", ["-NoProfile", "-NonInteractive", "-Command", ps_script])

	if exit_code != 0:
		_ps_print("[PC9] Ошибка запуска проводника Windows (код %d).\n" % exit_code)
		return

	if not FileAccess.file_exists(tmp_path):
		_ps_print("[PC9] Файл не выбран.\n")
		return

	var f_tmp : FileAccess = FileAccess.open(tmp_path, FileAccess.READ)
	var path  : String     = f_tmp.get_as_text().strip_edges()
	f_tmp.close()
	DirAccess.remove_absolute(tmp_path)

	if path.is_empty():
		_ps_print("[PC9] Файл не выбран.\n")
		return

	# PC10: перед фактическим запуском файл проходит через диалог
	# подтверждения PC10_Security_Bootloader (Admin PowerShell Guard)
	pc10_request_file_execution(path)

func _on_file_selected(path: String) -> void:
	var ext : String = path.get_extension().to_lower()
	var ok  : bool
	if ext == "com":
		ok = loader_load_com(path)
	else:
		ok = loader_load_exe(path)
	if ok:
		_ps_print("[PC9] Файл загружен. Запуск...\n")
		_start_emulation()

#  БЛОК 5 — CGA ДИСПЛЕЙ

var cga_image   : Image
var cga_texture : ImageTexture
var cga_rect    : TextureRect

func cga_init(parent: Control) -> void:
	cga_image   = Image.create(CGA_W, CGA_H, false, Image.FORMAT_RGBA8)
	cga_image.fill(Color(0, 0, 0))
	cga_texture = ImageTexture.create_from_image(cga_image)

	cga_rect = TextureRect.new()
	cga_rect.texture             = cga_texture
	cga_rect.expand_mode         = TextureRect.EXPAND_FIT_WIDTH_PROPORTIONAL
	cga_rect.stretch_mode        = TextureRect.STRETCH_SCALE
	cga_rect.custom_minimum_size = Vector2(CGA_W * 2, CGA_H * 2)
	cga_rect.size                = Vector2(CGA_W * 2, CGA_H * 2)
	parent.add_child(cga_rect)

func cga_render() -> void:
	if not cga_image: return
	for y in range(CGA_H):
		var row_base : int
		if y % 2 == 0:
			row_base = CGA_VRAM + (y / 2) * 80
		else:
			row_base = CGA_VRAM + 0x2000 + (y / 2) * 80

		for b in range(80):
			var addr : int = row_base + b
			var byte_val : int = memory[addr] if addr < RAM_SIZE else 0
			var x_base : int = b * 4
			cga_image.set_pixel(x_base + 0, y, CGA_PALETTE[(byte_val >> 6) & 3])
			cga_image.set_pixel(x_base + 1, y, CGA_PALETTE[(byte_val >> 4) & 3])
			cga_image.set_pixel(x_base + 2, y, CGA_PALETTE[(byte_val >> 2) & 3])
			cga_image.set_pixel(x_base + 3, y, CGA_PALETTE[byte_val & 3])

	cga_texture.update(cga_image)

# ══════════════════════════════════════════════════════════════
#  БЛОК 5B — ЗВУКОВАЯ КАРТА AdLib / OPL2 (новое в PC9)
# ══════════════════════════════════════════════════════════════
class OPL2Channel:
	var fnum       : int = 0
	var block      : int = 0
	var key_on     : bool = false
	var mult_mod   : float = 1.0   # Множитель частоты модулятора
	var mult_car   : float = 1.0   # Множитель частоты носителя
	var vol_mod    : float = 0.0   # Громкость модулятора (глубина FM)
	var vol_car    : float = 0.0   # Выходная громкость носителя
	var phase_mod  : float = 0.0   # Текущая фаза модулятора
	var phase_car  : float = 0.0   # Текущая фаза носителя

var adlib_player   : AudioStreamPlayer
var adlib_playback : AudioStreamGeneratorPlayback
var adlib_index    : int = 0
var adlib_regs     : PackedByteArray
var opl_channels   : Array = []

const SAMPLE_RATE  : float = 22050.0

# Карта соответствия смещений операторов OPL2 к 9 каналам эмулятора
const OP_TO_CHANNEL : Dictionary = {
	0x00: 0, 0x01: 1, 0x02: 2, 0x03: 0, 0x04: 1, 0x05: 2,
	0x08: 3, 0x09: 4, 0x0A: 5, 0x0B: 3, 0x0C: 4, 0x0D: 5,
	0x10: 6, 0x11: 7, 0x12: 8, 0x13: 6, 0x14: 7, 0x15: 8
}

# Определяем, является ли оператор модулятором (true) или носителем (false)
const OP_IS_MODULATOR : Dictionary = {
	0x00: true,  0x01: true,  0x02: true,  0x03: false, 0x04: false, 0x05: false,
	0x08: true,  0x09: true,  0x0A: true,  0x0B: false, 0x0C: false, 0x0D: false,
	0x10: true,  0x11: true,  0x12: true,  0x13: false, 0x14: false, 0x15: false
}

func adlib_init() -> void:
	adlib_regs.resize(256)
	adlib_regs.fill(0)

	opl_channels.clear()
	for i in range(9):
		opl_channels.append(OPL2Channel.new())

	adlib_player = AudioStreamPlayer.new()
	var generator : AudioStreamGenerator = AudioStreamGenerator.new()
	generator.mix_rate = SAMPLE_RATE
	generator.buffer_length = 0.08   # Буфер 80мс для предотвращения задержек/заиканий
	adlib_player.stream = generator
	add_child(adlib_player)
	adlib_player.play()
	adlib_playback = adlib_player.get_stream_playback()
	print("[SND] Звуковая карта AdLib OPL2 инициализирована.")

func adlib_write_register(reg: int, val: int) -> void:
	adlib_regs[reg] = val

	# 0x20 - 0x35: Множители частоты операторов (MULT)
	if reg >= 0x20 and reg <= 0x35:
		var op_offset : int = reg - 0x20
		if OP_TO_CHANNEL.has(op_offset):
			var ch_idx : int = OP_TO_CHANNEL[op_offset]
			var mult : float = float(val & 0x0F)
			if mult == 0: mult = 0.5
			if OP_IS_MODULATOR[op_offset]:
				opl_channels[ch_idx].mult_mod = mult
			else:
				opl_channels[ch_idx].mult_car = mult

	# 0x40 - 0x55: Уровень затухания / Громкость (Total Level)
	elif reg >= 0x40 and reg <= 0x55:
		var op_offset2 : int = reg - 0x40
		if OP_TO_CHANNEL.has(op_offset2):
			var ch_idx2 : int = OP_TO_CHANNEL[op_offset2]
			var tl  : int = val & 0x3F   # 6 бит уровня (0 = максимальная громкость)
			var vol : float = 1.0 - (tl / 63.0)
			if OP_IS_MODULATOR[op_offset2]:
				opl_channels[ch_idx2].vol_mod = vol * 8.0   # Усиливаем для глубины модуляции
			else:
				opl_channels[ch_idx2].vol_car = vol

	# 0xA0 - 0xA8: F-Number Low (младшие 8 бит частоты ноты для каналов 0-8)
	elif reg >= 0xA0 and reg <= 0xA8:
		var ch_idx3 : int = reg - 0xA0
		opl_channels[ch_idx3].fnum = (opl_channels[ch_idx3].fnum & 0x300) | val

	# 0xB0 - 0xB8: Key On (старт ноты), Block (октава), старшие биты F-Number
	elif reg >= 0xB0 and reg <= 0xB8:
		var ch_idx4 : int = reg - 0xB0
		opl_channels[ch_idx4].fnum  = (opl_channels[ch_idx4].fnum & 0xFF) | ((val & 3) << 8)
		opl_channels[ch_idx4].block = (val >> 2) & 7

		var new_key_on : bool = bool((val >> 5) & 1)
		if new_key_on != opl_channels[ch_idx4].key_on:
			opl_channels[ch_idx4].key_on = new_key_on
			if new_key_on:
				opl_channels[ch_idx4].phase_mod = 0.0
				opl_channels[ch_idx4].phase_car = 0.0

func adlib_update_audio() -> void:
	if adlib_playback == null:
		return

	var frames_to_fill : int = adlib_playback.get_frames_available()
	var dt : float = 1.0 / SAMPLE_RATE

	while frames_to_fill > 0:
		var chunk : int = mini(frames_to_fill, 128)
		for i in range(chunk):
			var mixed_sample : float = 0.0

			for ch_idx in range(9):
				var ch : OPL2Channel = opl_channels[ch_idx]
				if not ch.key_on or ch.fnum == 0:
					continue

				# Формула частоты OPL2: F = (FNum * 49716) / 2^(20 - Block)
				var base_freq : float = float(ch.fnum) * 49716.0 / float(1 << (20 - ch.block))
				if base_freq <= 0: continue

				# Оператор 1 (модулятор частоты)
				ch.phase_mod += 2.0 * PI * (base_freq * ch.mult_mod) * dt
				if ch.phase_mod > 2.0 * PI: ch.phase_mod -= 2.0 * PI
				var mod_out : float = sin(ch.phase_mod) * ch.vol_mod

				# Оператор 2 (носитель звука с частотной модуляцией)
				ch.phase_car += 2.0 * PI * (base_freq * ch.mult_car) * dt
				if ch.phase_car > 2.0 * PI: ch.phase_car -= 2.0 * PI

				var car_out : float = sin(ch.phase_car + mod_out) * ch.vol_car
				mixed_sample += car_out * 0.15

			mixed_sample = clamp(mixed_sample, -1.0, 1.0)
			adlib_playback.push_frame(Vector2(mixed_sample, mixed_sample))

		frames_to_fill -= chunk

# ── Системный таймер: обновление BIOS-счётчика тиков ─────────
func _tick_timer() -> void:
	cpu_handle_interrupt(0x08)   # PC9: тик идёт через настоящий IRQ0 (INT 08h)

func _tick_timer_raw() -> void:
	var addr : int = TIMER_ADDR
	if addr + 3 < RAM_SIZE:
		var val : int = (memory[addr] | (memory[addr+1] << 8)
			| (memory[addr+2] << 16) | (memory[addr+3] << 24))
		val = (val + 1) & 0xFFFFFFFF
		memory[addr]   = val & 0xFF
		memory[addr+1] = (val >> 8)  & 0xFF
		memory[addr+2] = (val >> 16) & 0xFF
		memory[addr+3] = (val >> 24) & 0xFF
		if val >= 1573040:   # PC11: сутки прошли (24ч * 60 * 60 * 18.2 тика/сек)
			memory[addr]   = 0
			memory[addr+1] = 0
			memory[addr+2] = 0
			memory[addr+3] = 0
			if PC11_BDA_TIMER_ROLL < RAM_SIZE:
				memory[PC11_BDA_TIMER_ROLL] = 1
	pc11_fire_timer_tick_hook()   # PC11: пользовательский INT 1Ch на каждый тик IRQ0

const PC11_BDA_TIMER_ROLL : int = 0x0470   # PC11: флаг перехода через полночь

func pc11_fire_timer_tick_hook() -> void:
	# В реальном BIOS обработчик INT 08h всегда дополнительно вызывает
	# INT 1Ch — "крюк" для программ, которые хотят выполняться по таймеру
	# (музыкальные плееры, анимация часов) без переопределения INT 08h.
	cpu_handle_interrupt(0x1C)

	ps_output_buf += text

func _ps_on_enter(cmd_raw: String) -> void:
	var cmd : String = cmd_raw.strip_edges()
	ps_input.clear()
	_ps_print(cmd + "\n")

	if ps_waiting_yn:
		var c : String = cmd.to_upper()
		if c in ["Y", "Д"]:
			_ps_print("[PC9] Открываем проводник...\n")
			ps_waiting_yn = false
			loader_open_file_dialog()
		elif c in ["N", "Н"]:
			_ps_print("[PC9] Отмена.\n")
			ps_waiting_yn = false
		else:
			_ps_print("Введите Y или N: ")
		return

	# PC10: пошаговый диалог входа/регистрации аккаунта
	if pc10_ps_auth_stage != "":
		match pc10_ps_auth_stage:
			"login_user":
				pc10_ps_auth_username = cmd
				pc10_ps_auth_stage    = "login_pass"
				_ps_print("Пароль: ")
			"login_pass":
				if pc10_login_user(pc10_ps_auth_username, cmd):
					_ps_print("[AUTH] Добро пожаловать, %s!\n" % pc10_ps_auth_username)
				else:
					_ps_print("[AUTH] Неверный логин или пароль.\n")
				pc10_ps_auth_stage    = ""
				pc10_ps_auth_username = ""
				_ps_print("\nPS C:\\PC10> ")
			"reg_user":
				pc10_ps_auth_username = cmd
				pc10_ps_auth_stage    = "reg_pass"
				_ps_print("Новый пароль: ")
			"reg_pass":
				if pc10_register_user(pc10_ps_auth_username, cmd):
					_ps_print("[AUTH] Аккаунт '%s' создан. Используйте LOGIN для входа.\n" % pc10_ps_auth_username)
				else:
					_ps_print("[AUTH] Аккаунт с таким именем уже существует.\n")
				pc10_ps_auth_stage    = ""
				pc10_ps_auth_username = ""
				_ps_print("\nPS C:\\PC10> ")
		return

	var cmd_up : String = cmd.to_upper()

	# PC9: сначала пробуем расширенные команды (демо/отладка)
	if _ps_extended_command(cmd_up):
		_ps_print("\nPS C:\\PC10> ")
		return

	match cmd_up:
		"RUN", "LOAD", "GAME", "START GAME":
			_ps_print("Открыть проводник для выбора .COM/.EXE? (Y/N): ")
			ps_waiting_yn = true
		"LOGIN":
			pc10_ps_auth_stage = "login_user"
			_ps_print("Имя пользователя: ")
		"REGISTER":
			pc10_ps_auth_stage = "reg_user"
			_ps_print("Имя нового пользователя: ")
		"LOGOUT":
			pc10_logout_user()
			_ps_print("[AUTH] Выход выполнен.\n")
		"WHOAMI":
			if pc10_current_user.is_empty():
				_ps_print("[AUTH] Сессия не авторизована.\n")
			else:
				_ps_print("[AUTH] Текущий пользователь: %s\n" % pc10_current_user)
		"VGA", "VGATEXT":
			pc10_open_vga_text_window()
		"TIGA":
			pc10_open_tiga_window()
		"REGS":
			var r : Dictionary = cpu_regs
			var s : Dictionary = cpu_segs
			_ps_print("AX=%04X BX=%04X CX=%04X DX=%04X\n" % [r["AX"],r["BX"],r["CX"],r["DX"]])
			_ps_print("SI=%04X DI=%04X SP=%04X BP=%04X\n" % [r["SI"],r["DI"],r["SP"],r["BP"]])
			_ps_print("CS=%04X DS=%04X SS=%04X ES=%04X\n" % [s["CS"],s["DS"],s["SS"],s["ES"]])
			_ps_print("IP=%04X  HALT=%s\n" % [cpu_ip, str(cpu_halt)])
		"FLAGS":
			_ps_print("CF=%d ZF=%d SF=%d OF=%d PF=%d IF=%d DF=%d AF=%d\n" % [
				cpu_flags["CF"], cpu_flags["ZF"], cpu_flags["SF"],
				cpu_flags["OF"], cpu_flags["PF"], cpu_flags["IF"],
				cpu_flags["DF"], cpu_flags["AF"]])
		"STOP":
			emu_running = false
			cpu_halt    = true
			_ps_print("[PC9] Эмулятор остановлен.\n")
		"RESET":
			cpu_halt    = true
			emu_running = false
			for k in cpu_regs.keys(): cpu_regs[k] = 0
			cpu_regs["SP"] = 0xFFFE
			cpu_ip = 0x0100
			_ps_print("[PC9] Сброс процессора.\n")
		"MEMTEST":
			_ps_print("[PC9] ОЗУ: %d KB / %d KB\n" % [
				memory.size() / 1024, RAM_SIZE / 1024])
		"CLS":
			ps_output_buf = ""
			if ps_text: ps_text.text = ""
		"HELP", "?":
			_ps_print(
				"Команды:\n"
				+ "  RUN / LOAD  — загрузить .COM или .EXE файл (с подтверждением\n"
				+ "                запуска через PC10_Security_Bootloader)\n"
				+ "  REGS        — показать регистры CPU\n"
				+ "  FLAGS       — показать флаги CPU\n"
				+ "  STOP        — остановить эмулятор\n"
				+ "  RESET       — сбросить CPU\n"
				+ "  MEMTEST     — проверить ОЗУ\n"
				+ "  CLS         — очистить экран\n"

var emu_running      : bool  = false
var emu_steps_budget : int   = 3000
var emu_frame_ms     : float = 0.0
const EMU_FRAME_INTERVAL : float = 0.0167   # ~60 Гц обновление экрана/звука

func _run_game() -> void:
	if ui_start_visible: _toggle_start_menu()
	loader_open_file_dialog()

func _start_emulation() -> void:
	emu_running = true
	_open_cga_window()

func _process(delta: float) -> void:
	# PC9: звук обновляется каждый кадр независимо от состояния CPU,
	# чтобы не было щелчков/обрывов в AdLib-плеере
	adlib_update_audio()

	# PC10: текстовый VGA-дисплей обновляется независимо от состояния
	# CPU, чтобы курсор продолжал мигать даже на паузе эмулятора
	pc10_vga_text_process(delta)

	if not emu_running or cpu_halt:
		if cpu_halt and emu_running:
			emu_running = false
			print("[EMU] Программа остановлена.")
		return

	emu_frame_ms += delta
	if emu_frame_ms < EMU_FRAME_INTERVAL:
		return
	emu_frame_ms = 0.0

	for _i in range(emu_steps_budget):
		if cpu_halt: break
		cpu_step()

	if cga_window and is_instance_valid(cga_window):
		cga_render()
	if tiga_window and is_instance_valid(tiga_window):
		tiga_update_hardware_frame()

const KEY_MAP : Dictionary = {
	KEY_SPACE:     [0x20, 0x39],
	KEY_ENTER:     [0x0D, 0x1C],
	KEY_ESCAPE:    [0x1B, 0x01],
	KEY_BACKSPACE: [0x08, 0x0E],
	KEY_TAB:       [0x09, 0x0F],
	KEY_UP:        [0x00, 0x48],
	KEY_DOWN:      [0x00, 0x50],
	KEY_LEFT:      [0x00, 0x4B],
	KEY_RIGHT:     [0x00, 0x4D],
	KEY_F1:        [0x00, 0x3B],
	KEY_F2:        [0x00, 0x3C],
	KEY_F3:        [0x00, 0x3D],
	KEY_F4:        [0x00, 0x3E],
	KEY_F5:        [0x00, 0x3F],
	KEY_F10:       [0x00, 0x44],
}

const BDA_KBD_FLAG1 : int = 0x0417   # PC11: статус Shift/Ctrl/Alt/Caps (как в AdvancedKeyboard)
const BDA_KBD_FLAG2 : int = 0x0418   # PC11: расширенный статус (левый/правый Alt/Ctrl)

func pc11_update_shift_flags(event: InputEventKey) -> void:
	var flags : int = memory[BDA_KBD_FLAG1]
	if event.keycode == KEY_SHIFT:
		flags = (flags | 0x03) if event.pressed else (flags & ~0x03)
	elif event.keycode == KEY_CTRL:
		flags = (flags | 0x04) if event.pressed else (flags & ~0x04)
	elif event.keycode == KEY_ALT:
		flags = (flags | 0x08) if event.pressed else (flags & ~0x08)
	elif event.keycode == KEY_CAPSLOCK and event.pressed:
		flags = flags ^ 0x40
	memory[BDA_KBD_FLAG1] = flags & 0xFF

func _input(event: InputEvent) -> void:
	if event is InputEventMouseMotion or event is InputEventMouseButton:
		pc11_mouse_handle_input(event)
		return
	if not event is InputEventKey: return
	var e : InputEventKey = event as InputEventKey
	pc11_update_shift_flags(e)   # PC11: обновляем биты Shift/Ctrl/Alt в BDA при каждом нажатии/отпускании
	if not event.pressed: return
	if KEY_MAP.has(e.keycode):
		var pair : Array = KEY_MAP[e.keycode]   # [ascii, scan]
		pc11_push_to_bios_buffer(pair[1], pair[0])
	elif e.unicode > 0:
		var c : int = e.unicode & 0x7F
		pc11_push_to_bios_buffer(c, c)

# ──────────────────────────────────────────────────────────────
#  PC11: МОБИЛЬНЫЙ ЭКРАННЫЙ ВВОД (D-pad / Fire / Esc для Android)
#  Кнопки на экране отправляют те же скан-коды, что и физическая
#  клавиатура, прямо в настоящий BDA-буфер — для CPU разницы нет.
# ──────────────────────────────────────────────────────────────
const PC11_MOBILE_KEY_MAP : Dictionary = {
	"btn_up":    [0x00, 0x48],
	"btn_down":  [0x00, 0x50],
	"btn_left":  [0x00, 0x4B],
	"btn_right": [0x00, 0x4D],
	"btn_fire":  [0x0D, 0x1C],
	"btn_esc":   [0x1B, 0x01],
}

var pc11_mobile_held_buttons : Dictionary = {}   # button_name -> true, пока палец держит кнопку

func pc11_mobile_button_pressed(button_name : String) -> void:
	if not PC11_MOBILE_KEY_MAP.has(button_name):
		return
	pc11_mobile_held_buttons[button_name] = true
	var pair2 : Array = PC11_MOBILE_KEY_MAP[button_name]   # [ascii, scan]
	pc11_push_to_bios_buffer(pair2[1], pair2[0])

func pc11_mobile_button_released(button_name : String) -> void:
	pc11_mobile_held_buttons.erase(button_name)

func pc11_mobile_process_held(_delta : float) -> void:
	for button_name in pc11_mobile_held_buttons.keys():
		if PC11_MOBILE_KEY_MAP.has(button_name):
			var pair3 : Array = PC11_MOBILE_KEY_MAP[button_name]
			pc11_push_to_bios_buffer(pair3[1], pair3[0])

# ══════════════════════════════════════════════════════════════
#  PC11 — МЫШЬ PS/2 (Microsoft Mouse API, INT 33h)
#  Реальная мышь хоста (тач/курсор Godot) транслируется в события
#  PS/2-мыши классического DOS: позиция в пикселях, состояние
#  кнопок, область видимости (clip) и пользовательский обработчик
#  событий (event handler), как у настоящего MOUSE.COM/MOUSE.SYS.
# ══════════════════════════════════════════════════════════════

var pc11_mouse_installed   : bool = false
var pc11_mouse_visible     : bool = false
var pc11_mouse_show_count  : int = 0           # счётчик ShowCursor/HideCursor (может быть отрицательным)
var pc11_mouse_x           : int = 320
var pc11_mouse_y           : int = 100
var pc11_mouse_buttons     : int = 0           # бит 0 = ЛКМ, бит 1 = ПКМ, бит 2 = СКМ
var pc11_mouse_min_x       : int = 0
var pc11_mouse_max_x       : int = 639
var pc11_mouse_min_y       : int = 0
var pc11_mouse_max_y       : int = 199
var pc11_mouse_last_press_x : int = 0
var pc11_mouse_last_press_y : int = 0
var pc11_mouse_last_release_x : int = 0
var pc11_mouse_last_release_y : int = 0
var pc11_mouse_sensitivity : float = 1.0       # масштаб экранных координат хоста -> координат DOS
var pc11_mouse_handler_seg : int = 0           # CS пользовательского обработчика событий (AX=0Ch)
var pc11_mouse_handler_off : int = 0           # IP пользовательского обработчика событий
var pc11_mouse_handler_mask : int = 0          # маска событий, на которые подписан обработчик

func pc11_mouse_handle_input(event : InputEvent) -> void:
	if not pc11_mouse_installed:
		return
	if event is InputEventMouseMotion:
		var m : InputEventMouseMotion = event as InputEventMouseMotion
		pc11_mouse_x = clampi(int(m.position.x * pc11_mouse_sensitivity), pc11_mouse_min_x, pc11_mouse_max_x)
		pc11_mouse_y = clampi(int(m.position.y * pc11_mouse_sensitivity), pc11_mouse_min_y, pc11_mouse_max_y)
		pc11_mouse_fire_event_handler(0x01)   # бит 0 маски — движение мыши
	elif event is InputEventMouseButton:
		var b : InputEventMouseButton = event as InputEventMouseButton
		var bit : int = 0
		match b.button_index:
			MOUSE_BUTTON_LEFT:   bit = 0x01
			MOUSE_BUTTON_RIGHT:  bit = 0x02
			MOUSE_BUTTON_MIDDLE: bit = 0x04
		if bit == 0:
			return
		if b.pressed:
			pc11_mouse_buttons |= bit
			pc11_mouse_last_press_x = pc11_mouse_x
			pc11_mouse_last_press_y = pc11_mouse_y
			pc11_mouse_fire_event_handler(bit << 1)        # биты 1/3/5 — нажатие
		else:
			pc11_mouse_buttons &= ~bit
			pc11_mouse_last_release_x = pc11_mouse_x
			pc11_mouse_last_release_y = pc11_mouse_y
			pc11_mouse_fire_event_handler(bit << 2)        # биты 2/4/6 — отпускание

func pc11_mouse_fire_event_handler(event_mask : int) -> void:
	# Полноценный вызов реального обработчика потребовал бы прямого вмешательства
	# в стек CPU (far call с передачей AX/BX/CX/DX как в настоящем MOUSE.SYS).
	# Здесь делаем безопасную HLE-версию: если обработчик зарегистрирован и его
	# маска совпадает с произошедшим событием, кладём данные в регистры и просто
	# логируем вызов — полноценная передача управления того же типа, что и для
	# TIGA HLE API из базового PC10.
	if pc11_mouse_handler_seg == 0 and pc11_mouse_handler_off == 0:
		return
	if (pc11_mouse_handler_mask & event_mask) == 0:
		return
	if pc11_logger_verbose:
		print("[PC11 MOUSE] Событие 0x%02X -> обработчик %04X:%04X" % [
			event_mask, pc11_mouse_handler_seg, pc11_mouse_handler_off])

func _int_33h() -> void:
	var ax : int = cpu_regs["AX"]
	match ax:
		0x00:   # Сброс/установка драйвера мыши
			pc11_mouse_installed = true
			pc11_mouse_visible = false
			pc11_mouse_show_count = 0
			pc11_mouse_buttons = 0
			pc11_mouse_x = (pc11_mouse_min_x + pc11_mouse_max_x) / 2
			pc11_mouse_y = (pc11_mouse_min_y + pc11_mouse_max_y) / 2
			cpu_regs["AX"] = 0xFFFF   # драйвер установлен
			cpu_regs["BX"] = 3        # количество кнопок
		0x01:   # Показать курсор
			pc11_mouse_show_count += 1
			pc11_mouse_visible = pc11_mouse_show_count >= 0
		0x02:   # Скрыть курсор
			pc11_mouse_show_count -= 1
			pc11_mouse_visible = pc11_mouse_show_count >= 0
		0x03:   # Получить позицию и состояние кнопок
			cpu_regs["BX"] = pc11_mouse_buttons
			cpu_regs["CX"] = pc11_mouse_x
			cpu_regs["DX"] = pc11_mouse_y
		0x04:   # Установить позицию курсора программно
			pc11_mouse_x = clampi(cpu_regs["CX"], pc11_mouse_min_x, pc11_mouse_max_x)
			pc11_mouse_y = clampi(cpu_regs["DX"], pc11_mouse_min_y, pc11_mouse_max_y)
		0x05:   # Получить данные о нажатии кнопки (счётчик + позиция последнего нажатия)
			var btn_idx : int = cpu_regs["BX"]
			cpu_regs["BX"] = 0   # счётчик нажатий не отслеживаем подробно — упрощённая реализация
			if btn_idx == 0:
				cpu_regs["CX"] = pc11_mouse_last_press_x
				cpu_regs["DX"] = pc11_mouse_last_press_y
		0x06:   # Получить данные об отпускании кнопки
			var btn_idx2 : int = cpu_regs["BX"]
			cpu_regs["BX"] = 0
			if btn_idx2 == 0:
				cpu_regs["CX"] = pc11_mouse_last_release_x
				cpu_regs["DX"] = pc11_mouse_last_release_y
		0x07:   # Установить горизонтальную область (min CX, max DX)
			pc11_mouse_min_x = mini(cpu_regs["CX"], cpu_regs["DX"])
			pc11_mouse_max_x = maxi(cpu_regs["CX"], cpu_regs["DX"])
			pc11_mouse_x = clampi(pc11_mouse_x, pc11_mouse_min_x, pc11_mouse_max_x)
		0x08:   # Установить вертикальную область (min CX, max DX)
			pc11_mouse_min_y = mini(cpu_regs["CX"], cpu_regs["DX"])
			pc11_mouse_max_y = maxi(cpu_regs["CX"], cpu_regs["DX"])
			pc11_mouse_y = clampi(pc11_mouse_y, pc11_mouse_min_y, pc11_mouse_max_y)
		0x0B:   # Получить счётчики смещения мыши (motion counters) с последнего вызова
			cpu_regs["CX"] = 0
			cpu_regs["DX"] = 0
		0x0C:   # Установить пользовательский обработчик событий мыши
			pc11_mouse_handler_mask = cpu_regs["CX"]
			pc11_mouse_handler_seg  = cpu_segs["ES"]
			pc11_mouse_handler_off  = cpu_regs["DX"]
		0x0F:   # Установить чувствительность мыши (горизонт./вертик.)
			var raw_sens : int = cpu_regs["CX"]
			pc11_mouse_sensitivity = max(0.1, float(raw_sens) / 8.0)
		0x13:   # Установить порог двойного клика (игнорируем, но не выдаём ошибку)
			pass
		0x21:   # Программный аппаратный сброс мыши
			pc11_mouse_installed = true
			cpu_regs["AX"] = 0xFFFF
		_:
			if pc11_logger_verbose:
				print("[PC11 MOUSE] Нереализованная подфункция AX=0x%04X" % ax)

func pc11_mouse_get_cursor_screen_pos() -> Vector2:
	# Удобный геттер для UI-слоя (например, VGATextDisplay), который хочет
	# нарисовать курсор мыши поверх текстового экрана — переводит координаты
	# DOS-мыши обратно в пиксели сцены с учётом текущей чувствительности.
	if pc11_mouse_sensitivity <= 0.0:
		return Vector2(pc11_mouse_x, pc11_mouse_y)
	return Vector2(pc11_mouse_x, pc11_mouse_y) / pc11_mouse_sensitivity

# ══════════════════════════════════════════════════════════════
#  PC11 — ПОСЛЕДОВАТЕЛЬНЫЙ ПОРТ COM1 (UART 8250/16450, INT 14h)
#  Базовые порты 0x3F8-0x3FF, как на настоящем PC XT/AT. Реализован
#  программный loopback по умолчанию (то, что программа отправляет,
#  она же и получает обратно через приёмный буфер) — удобно для
#  тестирования терминальных программ без реального второго ПК.
#  При необходимости loopback можно выключить и подключить порт
#  к внешнему источнику данных (pc11_uart_inject_byte).
# ══════════════════════════════════════════════════════════════

const PC11_UART_BASE       : int = 0x3F8   # COM1
const PC11_REG_THR_RBR     : int = 0       # Transmit Holding / Receive Buffer (DLAB=0)
const PC11_REG_DLL         : int = 0       # Divisor Latch Low  (DLAB=1)
const PC11_REG_IER         : int = 1       # Interrupt Enable   (DLAB=0)
const PC11_REG_DLH         : int = 1       # Divisor Latch High (DLAB=1)
const PC11_REG_IIR         : int = 2       # Interrupt Identification (чтение)
const PC11_REG_FCR         : int = 2       # FIFO Control (запись)
const PC11_REG_LCR         : int = 3       # Line Control (бит 7 = DLAB)
const PC11_REG_MCR         : int = 4       # Modem Control
const PC11_REG_LSR         : int = 5       # Line Status
const PC11_REG_MSR         : int = 6       # Modem Status
const PC11_REG_SCRATCH     : int = 7       # Scratch Register

const PC11_LSR_DATA_READY  : int = 0x01
const PC11_LSR_THRE        : int = 0x20    # Transmit Holding Register Empty
const PC11_LSR_TX_IDLE     : int = 0x40

var pc11_uart_loopback     : bool = true   # PC11: по умолчанию включён программный loopback
var pc11_uart_rx_buffer    : Array = []    # очередь принятых байт (FIFO)
var pc11_uart_divisor_low  : int = 0x0C    # делитель скорости (по умолчанию ~9600 бод при 1.8432МГц)
var pc11_uart_divisor_high : int = 0x00
var pc11_uart_lcr          : int = 0x03    # 8 бит данных, без чётности, 1 стоп-бит
var pc11_uart_mcr          : int = 0x00
var pc11_uart_ier          : int = 0x00
var pc11_uart_last_tx_byte : int = 0
var pc11_uart_tx_log       : PackedByteArray = PackedByteArray()   # история отправленных байт (для отладки/PowerShell)

func pc11_uart_dlab() -> bool:
	return (pc11_uart_lcr & 0x80) != 0

func pc11_uart_port_read(port : int) -> int:
	var reg : int = port - PC11_UART_BASE
	match reg:
		PC11_REG_THR_RBR:
			if pc11_uart_dlab():
				return pc11_uart_divisor_low
			if pc11_uart_rx_buffer.size() > 0:
				return pc11_uart_rx_buffer.pop_front()
			return 0x00
		PC11_REG_IER:
			if pc11_uart_dlab():
				return pc11_uart_divisor_high
			return pc11_uart_ier
		PC11_REG_IIR:
			return 0x01   # нет ожидающих прерываний (упрощённая реализация без IRQ4)
		PC11_REG_LCR:
			return pc11_uart_lcr
		PC11_REG_MCR:
			return pc11_uart_mcr
		PC11_REG_LSR:
			var lsr : int = PC11_LSR_THRE | PC11_LSR_TX_IDLE
			if pc11_uart_rx_buffer.size() > 0:
				lsr |= PC11_LSR_DATA_READY
			return lsr
		PC11_REG_MSR:
			return 0xB0   # CTS=1, DSR=1, DCD=1 — имитируем подключённое устройство
		PC11_REG_SCRATCH:
			return 0x00
		_:
			return 0xFF

func pc11_uart_port_write(port : int, val : int) -> void:
	var reg : int = port - PC11_UART_BASE
	var v : int = val & 0xFF
	match reg:
		PC11_REG_THR_RBR:
			if pc11_uart_dlab():
				pc11_uart_divisor_low = v
			else:
				pc11_uart_transmit_byte(v)
		PC11_REG_IER:
			if pc11_uart_dlab():
				pc11_uart_divisor_high = v
			else:
				pc11_uart_ier = v
		PC11_REG_FCR:
			if v & 0x02:
				pc11_uart_rx_buffer.clear()   # сброс приёмного FIFO
		PC11_REG_LCR:
			pc11_uart_lcr = v
		PC11_REG_MCR:
			pc11_uart_mcr = v
		PC11_REG_SCRATCH:
			pass
		_:
			pass

func pc11_uart_transmit_byte(b : int) -> void:
	pc11_uart_last_tx_byte = b
	pc11_uart_tx_log.append(b)
	if pc11_uart_tx_log.size() > 4096:
		pc11_uart_tx_log = pc11_uart_tx_log.slice(pc11_uart_tx_log.size() - 4096)
	if pc11_uart_loopback:
		pc11_uart_inject_byte(b)

func pc11_uart_inject_byte(b : int) -> void:
	# Кладёт байт во входящий буфер так, будто его прислал внешний COM-порт.
	# Вызывается loopback'ом автоматически, либо вручную при подключении
	# реального источника данных (сокет, Bluetooth-SPP, второй эмулятор).
	pc11_uart_rx_buffer.append(b & 0xFF)
	if pc11_uart_rx_buffer.size() > 4096:
		pc11_uart_rx_buffer.pop_front()

func pc11_uart_get_baud_rate() -> int:
	var divisor : int = (pc11_uart_divisor_high << 8) | pc11_uart_divisor_low
	if divisor <= 0:
		return 0
	return int(115200.0 / divisor)

func pc11_uart_get_tx_log_text() -> String:
	var s : String = ""
	for b in pc11_uart_tx_log:
		s += char(b)
	return s

# ── INT 14h — Последовательный порт BIOS (COM1) ───────────────
func _int_14h() -> void:
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	var al : int =  cpu_regs["AX"]       & 0xFF
	match ah:
		0x00:   # Инициализация порта: AL = параметры (биты скорости/чётности/стопа/длины)
			var baud_idx : int = (al >> 5) & 0x07
			const BAUD_DIVISORS : Array = [2304, 1536, 768, 384, 192, 96, 48, 24]
			pc11_uart_divisor_low  = BAUD_DIVISORS[baud_idx] & 0xFF
			pc11_uart_divisor_high = (BAUD_DIVISORS[baud_idx] >> 8) & 0xFF
			pc11_uart_lcr = al & 0x1F
			cpu_regs["AX"] = (0x60 << 8) | (al)   # AH = статус линии (готов к передаче)
		0x01:   # Отправить байт: AL = байт для отправки
			pc11_uart_transmit_byte(al)
			var lsr_status : int = PC11_LSR_THRE | PC11_LSR_TX_IDLE
			cpu_regs["AX"] = (lsr_status << 8) | al
		0x02:   # Принять байт: возвращает AL = принятый байт
			if pc11_uart_rx_buffer.size() > 0:
				var b2 : int = pc11_uart_rx_buffer.pop_front()
				cpu_regs["AX"] = (0x00 << 8) | b2
			else:
				cpu_regs["AX"] = (0x80 << 8) | 0x00   # AH бит 7 = тайм-аут, данных нет
		0x03:   # Получить статус порта: AH = Line Status, AL = Modem Status
			var lsr3 : int = PC11_LSR_THRE | PC11_LSR_TX_IDLE
			if pc11_uart_rx_buffer.size() > 0:
				lsr3 |= PC11_LSR_DATA_READY
			cpu_regs["AX"] = (lsr3 << 8) | 0xB0
		_:
			if pc11_logger_verbose:
				print("[PC11 COM] Нереализованная подфункция AH=0x%02X" % ah)

func debug_dump_regs() -> String:
	var r : Dictionary = cpu_regs
	var s : Dictionary = cpu_segs
	var f : Dictionary = cpu_flags
	return (
		"=== CPU STATE ===\n"
		+ "AX=%04X  BX=%04X  CX=%04X  DX=%04X\n" % [r["AX"],r["BX"],r["CX"],r["DX"]]
		+ "SI=%04X  DI=%04X  SP=%04X  BP=%04X\n" % [r["SI"],r["DI"],r["SP"],r["BP"]]
		+ "CS=%04X  DS=%04X  SS=%04X  ES=%04X\n" % [s["CS"],s["DS"],s["SS"],s["ES"]]
		+ "IP=%04X  HALT=%s\n" % [cpu_ip, str(cpu_halt)]
		+ "CF=%d ZF=%d SF=%d OF=%d PF=%d IF=%d DF=%d AF=%d\n" % [
			f["CF"],f["ZF"],f["SF"],f["OF"],f["PF"],f["IF"],f["DF"],f["AF"]]
	)

func debug_dump_stack(count: int = 8) -> String:
	var out : String = "=== STACK (SS:%04X) ===\n" % cpu_segs["SS"]
	var sp  : int    = cpu_regs["SP"]
	for i in range(count):
		var off  : int = (sp + i * 2) & 0xFFFF
		var val  : int = cpu_rw("SS", off)
		var mark : String = " <-- SP" if i == 0 else ""
		out += "  [%04X] = %04X%s\n" % [off, val, mark]
	return out

func debug_dump_mem(seg: int, off: int, count: int = 64) -> String:
	var out  : String = "=== MEM %04X:%04X ===\n" % [seg, off]
	var base : int    = mem_get_phys(seg, off)
	var i    : int    = 0
	while i < count:
		var line : String = "%04X: " % (off + i)
		var chars : String = ""
		for j in range(16):
			if i + j >= count: break
			var b : int = memory[base + i + j] if base + i + j < RAM_SIZE else 0
			line  += "%02X " % b
			chars += char(b) if b >= 0x20 and b < 0x7F else "."
		out += line + " " + chars + "\n"
		i += 16
	return out

func disasm_at(seg: int, off: int, lines: int = 10) -> String:
	var out  : String = "=== DISASM %04X:%04X ===\n" % [seg, off]
	var base : int    = mem_get_phys(seg, off)
	var pos  : int    = 0
	for _l in range(lines):
		if base + pos >= RAM_SIZE: break
		var b    : int    = memory[base + pos]
		var mnem : String = _disasm_mnem(b)
		out += "%04X: %02X  %s\n" % [off + pos, b, mnem]
		pos += 1
	return out

func demo_load_hello_world() -> void:
	# Минимальная .COM программа: выводит "Hello, PC9!" через INT 21h
	var prog : PackedByteArray = PackedByteArray([
		0xB4, 0x09,        # MOV AH, 09h
		0xBA, 0x08, 0x01,  # MOV DX, 0x0108  (смещение строки в COM)
		0xCD, 0x21,        # INT 21h
		0xCD, 0x20,        # INT 20h
		0x48, 0x65, 0x6C, 0x6C, 0x6F, 0x2C,   # "Hello,"
		0x20, 0x50, 0x43, 0x39, 0x21,          # " PC9!"
		0x24                                    # '$'
	])

	var seg  : int = LOAD_SEGMENT
	loader_create_psp(seg, "HELLO.COM")
	var base : int = (seg << 4) + PSP_SIZE
	for i in range(prog.size()):
		if base + i < RAM_SIZE:
			memory[base + i] = prog[i]

	cpu_segs["CS"] = seg; cpu_segs["DS"] = seg
	cpu_segs["ES"] = seg; cpu_segs["SS"] = seg
	cpu_ip         = 0x0100
	cpu_regs["SP"] = 0xFFFE
	cpu_halt       = false
	_ps_print("[DEMO] Hello World загружен.\n")

func hex_to_bytes(hex_str: String) -> PackedByteArray:
	var result : PackedByteArray = PackedByteArray()
	var s      : String = hex_str.replace(" ", "").replace("\n", "").to_upper()
	var i      : int    = 0
	while i + 1 < s.length():
		var hi  : int = _hex_nibble(s[i])
		var lo  : int = _hex_nibble(s[i + 1])
		if hi < 0 or lo < 0: i += 2; continue
		result.append((hi << 4) | lo)
		i += 2
	return result

func _hex_nibble(c: String) -> int:
	if c >= "0" and c <= "9": return c.unicode_at(0) - 0x30
	if c >= "A" and c <= "F": return c.unicode_at(0) - 0x37
	return -1

func byte_to_bin(b: int) -> String:
	var s : String = ""
	for i in range(7, -1, -1):
		s += "1" if (b >> i) & 1 else "0"
	return s

func mem_checksum(start: int, length: int) -> int:
	var sum : int = 0
	for i in range(length):
		if start + i < RAM_SIZE:
			sum = (sum + memory[start + i]) & 0xFFFF
	return sum

func mem_find(pattern: PackedByteArray, start: int = 0) -> int:
	var plen : int = pattern.size()
	if plen == 0: return -1
	for i in range(start, RAM_SIZE - plen):
		var found : bool = true
		for j in range(plen):
			if memory[i + j] != pattern[j]:
				found = false
				break
		if found: return i
	return -1

func mem_copy(src: int, dst: int, count: int) -> void:
	if src == dst: return
	for i in range(count):
		if src + i < RAM_SIZE and dst + i < RAM_SIZE:
			memory[dst + i] = memory[src + i]

func mem_fill(start: int, count: int, val: int) -> void:
	for i in range(count):
		if start + i < RAM_SIZE:
			memory[start + i] = val & 0xFF

# ══════════════════════════════════════════════════════════════
#  БЛОК 15 — PC10: VGATextDisplay (текстовый режим 80×25, Mode 3h)
#  Архитектура: высокопроизводительный рендерер текстовой VRAM,
#  кэширование изменений, 16-цветная DOS-палитра, аппаратный курсор.
#  В оригинале PC10 это был отдельный класс VGATextDisplay extends
#  Control; здесь он интегрирован прямо в главный скрипт и рисует
#  внутри своего собственного дочернего узла Control (vga_text_node),
#  читая ту же область VRAM (0xB8000), что и обычный CGA-блок PC9.
# ══════════════════════════════════════════════════════════════

var vga_text_node        : Control
var vga_text_char_cache  : PackedByteArray
var vga_text_attr_cache  : PackedByteArray
var vga_text_blink_timer : float = 0.0
var vga_text_blink_state : bool  = true
var vga_text_cell_w      : float = 8.0
var vga_text_cell_h      : float = 16.0
var vga_text_window      : Window

func pc10_vga_text_init() -> void:
	vga_text_char_cache = PackedByteArray()
	vga_text_attr_cache = PackedByteArray()
	vga_text_char_cache.resize(VGA_TEXT_CELLS)
	vga_text_attr_cache.resize(VGA_TEXT_CELLS)
	vga_text_char_cache.fill(0)
	vga_text_attr_cache.fill(0)

func pc10_vga_text_recalc_layout() -> void:
	if not vga_text_node: return
	vga_text_cell_w = vga_text_node.size.x / float(VGA_TEXT_COLS)
	vga_text_cell_h = vga_text_node.size.y / float(VGA_TEXT_ROWS)
	vga_text_node.queue_redraw()

func pc10_vga_text_process(_delta: float) -> void:
	if memory.is_empty():
		return
	vga_text_blink_timer += _delta
	if vga_text_blink_timer >= 0.25:
		vga_text_blink_timer = 0.0
		vga_text_blink_state = not vga_text_blink_state
		if vga_text_node: vga_text_node.queue_redraw()

	var is_dirty : bool = false
	for i in range(VGA_TEXT_CELLS):
		var vram_idx : int = VGA_TEXT_VRAM + (i * 2)
		if vram_idx + 1 >= memory.size():
			break
		var current_char : int = memory[vram_idx]
		var current_attr : int = memory[vram_idx + 1]
		if vga_text_char_cache[i] != current_char or vga_text_attr_cache[i] != current_attr:
			vga_text_char_cache[i] = current_char
			vga_text_attr_cache[i] = current_attr
			is_dirty = true
	if is_dirty and vga_text_node:
		vga_text_node.queue_redraw()

func pc10_vga_text_draw() -> void:
	if not vga_text_node or memory.is_empty():
		return
	for row in range(VGA_TEXT_ROWS):
		for col in range(VGA_TEXT_COLS):
			var cell_idx  : int = row * VGA_TEXT_COLS + col
			var char_code : int = vga_text_char_cache[cell_idx]
			var attr      : int = vga_text_attr_cache[cell_idx]

			var fg_color_idx : int  = attr & 0x0F
			var bg_color_idx : int  = (attr >> 4) & 0x0F
			var is_blinking  : bool = bool(attr & 0x80)
			if is_blinking:
				bg_color_idx &= 0x07

			var fg_color : Color = DOS_PALETTE_16[fg_color_idx]
			var bg_color : Color = DOS_PALETTE_16[bg_color_idx]
			if is_blinking and not vga_text_blink_state:
				fg_color = bg_color

			var pos_x : float = col * vga_text_cell_w
			var pos_y : float = row * vga_text_cell_h
			var rect  : Rect2 = Rect2(pos_x, pos_y, vga_text_cell_w, vga_text_cell_h)

			if bg_color_idx != 0:
				vga_text_node.draw_rect(rect, bg_color, true)

			if char_code > 32:
				var string_char : String = String.chr(char_code)
				var font_size   : int    = int(vga_text_cell_h * 0.9)
				var char_pos    : Vector2 = Vector2(pos_x, pos_y + (vga_text_cell_h * 0.82))
				vga_text_node.draw_string(ThemeDB.fallback_font, char_pos, string_char,
					HORIZONTAL_ALIGNMENT_LEFT, vga_text_cell_w, font_size, fg_color)

	pc10_vga_text_draw_cursor()
	pc11_vga_draw_mouse_cursor()   # PC11: курсор мыши PS/2 поверх текстового экрана

func pc10_vga_text_draw_cursor() -> void:
	if not vga_text_node: return
	if 0x451 >= memory.size(): return
	var cursor_col : int = memory[0x450]
	var cursor_row : int = memory[0x451]
	if cursor_col >= VGA_TEXT_COLS or cursor_row >= VGA_TEXT_ROWS:
		return
	if vga_text_blink_state:
		var c_x : float = cursor_col * vga_text_cell_w
		var c_y : float = cursor_row * vga_text_cell_h
		var cursor_start_line : int = memory[0x461] & 0x1F
		var cursor_end_line   : int = memory[0x460] & 0x1F
		var h_thickness : float = clamp(absf(float(cursor_end_line - cursor_start_line)), 2.0, vga_text_cell_h)
		var cursor_rect : Rect2 = Rect2(c_x, c_y + vga_text_cell_h - h_thickness, vga_text_cell_w, h_thickness)
		var cell_idx : int = cursor_row * VGA_TEXT_COLS + cursor_col
		var attr     : int = vga_text_attr_cache[cell_idx]
		var cursor_color : Color = DOS_PALETTE_16[attr & 0x0F]
		vga_text_node.draw_rect(cursor_rect, cursor_color, true)

func pc10_vga_text_demo_write(text: String, attr: int = 0x71, row: int = 0, col: int = 0) -> void:
	var addr : int = VGA_TEXT_VRAM + (row * VGA_TEXT_COLS + col) * 2
	for i in range(text.length()):
		if addr + 1 < RAM_SIZE:
			memory[addr]     = text.unicode_at(i)
			memory[addr + 1] = attr & 0xFF
			addr += 2
	memory[0x450] = col
	memory[0x451] = row + 1

#  БЛОК 16 — PC10: TIGADisplay / TIGACoprocessor
#  Ультра-разрешение 1280×1024, 256 цветов индексной палитры.
#  Архитектура: высокоуровневый акселератор (HLE TIGA Engine) +
#  отдельный программный сопроцессор с Host Interface Registers

var tiga_vram_buffer : PackedByteArray
var tiga_image        : Image
var tiga_texture       : ImageTexture
var tiga_rect          : TextureRect
var tiga_palette       : Array = []
var tiga_window        : Window

# Host Interface Registers сопроцессора TIGA (порты 0x380-0x384, теневые)
var tiga_hstadr_reg  : int = 0x00000000
var tiga_hstcntl_reg : int = 0x0000
var tiga_hstdata_reg : int = 0x0000
var tiga_fg_color    : int = 0x0F
var tiga_bg_color    : int = 0x00
var tiga_draw_mask   : int = 0xFF

enum PC10_ROP { COPY = 0, XOR = 1, AND = 2, OR = 3, NOT = 4 }
var tiga_current_rop : int = PC10_ROP.COPY

func pc10_tiga_init() -> void:
	tiga_vram_buffer = PackedByteArray()
	tiga_vram_buffer.resize(TIGA_WIDTH * TIGA_HEIGHT)
	tiga_vram_buffer.fill(0x00)

	tiga_palette.resize(256)
	for i in range(256):
		tiga_palette[i] = Color.from_hsv(float(i) / 256.0, 0.9, 0.9)

	tiga_image   = Image.create(TIGA_WIDTH, TIGA_HEIGHT, false, Image.FORMAT_RGBA8)
	tiga_texture = ImageTexture.create_from_image(tiga_image)

func tiga_hle_clear_screen(color_idx: int) -> void:
	tiga_vram_buffer.fill(color_idx & 0xFF)

func tiga_hle_set_pixel(x: int, y: int, color_idx: int) -> void:
	if x >= 0 and x < TIGA_WIDTH and y >= 0 and y < TIGA_HEIGHT:
		tiga_vram_buffer[y * TIGA_WIDTH + x] = color_idx & 0xFF

func tiga_hle_draw_line(x0: int, y0: int, x1: int, y1: int, color_idx: int) -> void:
	var dx : int = abs(x1 - x0)
	var dy : int = abs(y1 - y0)
	var sx : int = 1 if x0 < x1 else -1
	var sy : int = 1 if y0 < y1 else -1
	var err : int = dx - dy
	while true:
		tiga_hle_set_pixel(x0, y0, color_idx)
		if x0 == x1 and y0 == y1:
			break
		var e2 : int = 2 * err
		if e2 > -dy:
			err -= dy
			x0 += sx
		if e2 < dx:
			err += dx
			y0 += sy

func tiga_hle_fill_rect(x: int, y: int, w: int, h: int, color_idx: int) -> void:
	var c : int = color_idx & 0xFF
	for curr_y in range(y, y + h):
		if curr_y < 0 or curr_y >= TIGA_HEIGHT: continue
		var row_start : int = curr_y * TIGA_WIDTH
		for curr_x in range(x, x + w):
			if curr_x < 0 or curr_x >= TIGA_WIDTH: continue
			tiga_vram_buffer[row_start + curr_x] = c

func tiga_update_hardware_frame() -> void:
	if not tiga_image or not tiga_texture: return
	var rgba_bytes : PackedByteArray = PackedByteArray()
	rgba_bytes.resize(TIGA_WIDTH * TIGA_HEIGHT * 4)
	for i in range(TIGA_WIDTH * TIGA_HEIGHT):
		var color : Color = tiga_palette[tiga_vram_buffer[i]]
		var idx : int = i * 4
		rgba_bytes[idx]     = int(color.r * 255)
		rgba_bytes[idx + 1] = int(color.g * 255)
		rgba_bytes[idx + 2] = int(color.b * 255)
		rgba_bytes[idx + 3] = 255
	tiga_image.set_data(TIGA_WIDTH, TIGA_HEIGHT, false, Image.FORMAT_RGBA8, rgba_bytes)
	tiga_texture.update(tiga_image)

# ── HLE-диспетчер команд TIGA через "программное прерывание" AX=0x40xx ──
func pc10_handle_tiga_interrupt(ax_val: int) -> bool:
	match ax_val:
		0x4001:
			tiga_hle_clear_screen(cpu_regs["BX"])
			return true
		0x4002:
			tiga_hle_set_pixel(cpu_regs["CX"], cpu_regs["DX"], cpu_regs["BX"])
			return true
		0x400A:
			var segment  : int = cpu_segs["ES"]
			var offset   : int = cpu_regs["BX"]
			var phys_addr : int = (segment << 4) + offset
			if phys_addr + 7 >= RAM_SIZE: return true
			var x0 : int = memory[phys_addr]     | (memory[phys_addr+1] << 8)
			var y0 : int = memory[phys_addr+2]   | (memory[phys_addr+3] << 8)
			var x1 : int = memory[phys_addr+4]   | (memory[phys_addr+5] << 8)
			var y1 : int = memory[phys_addr+6]   | (memory[phys_addr+7] << 8)
			tiga_hle_draw_line(x0, y0, x1, y1, cpu_regs["CX"])
			return true
		0x400B:
			tiga_hle_fill_rect(cpu_regs["CX"], cpu_regs["DX"], cpu_regs["SI"], cpu_regs["DI"], cpu_regs["BX"])
			return true
	return false

# ── Аппаратный сопроцессор: запись/чтение VRAM через HSTADR/HSTDATA ──
func tiga_write_vram_at_hstadr(value: int) -> void:
	if tiga_vram_buffer.is_empty(): return
	var target_pixel : int = tiga_hstadr_reg % (TIGA_WIDTH * TIGA_HEIGHT)
	tiga_vram_buffer[target_pixel] = value & 0xFF
	if target_pixel + 1 < tiga_vram_buffer.size():
		tiga_vram_buffer[target_pixel + 1] = (value >> 8) & 0xFF

func tiga_read_vram_at_hstadr() -> int:
	if tiga_vram_buffer.is_empty(): return 0
	var target_pixel : int = tiga_hstadr_reg % (TIGA_WIDTH * TIGA_HEIGHT)
	var lo : int = tiga_vram_buffer[target_pixel]
	var hi : int = tiga_vram_buffer[(target_pixel + 1) % tiga_vram_buffer.size()]
	return lo | (hi << 8)

	for i in range(0, 800, 20):
		if mem_addr + 7 < RAM_SIZE:
			memory[mem_addr]   = (100 + i) & 0xFF
			memory[mem_addr+1] = ((100 + i) >> 8) & 0xFF
			memory[mem_addr+2] = 100 & 0xFF
			memory[mem_addr+3] = 0
			memory[mem_addr+4] = (1180 - i) & 0xFF
			memory[mem_addr+5] = ((1180 - i) >> 8) & 0xFF
			memory[mem_addr+6] = 924 & 0xFF
			memory[mem_addr+7] = (924 >> 8) & 0xFF

# ══════════════════════════════════════════════════════════════
#  БЛОК 17 — PC10: ATA/IDE КОНТРОЛЛЕР (порты 0x1F0 — 0x1F7)
#  Архитектура: register-level эмуляция жёсткого диска поверх
#  файла-образа на хост-машине, CHS-геометрия, статусные флаги
#  BSY/DRDY/DRQ/ERR, потоковая передача секторов через порт данных.
# ══════════════════════════════════════════════════════════════

var ide_reg_status      : int = IDE_STATUS_DRDY
var ide_reg_error       : int = 0
var ide_reg_sector_count: int = 0
var ide_reg_sector_num  : int = 1
var ide_reg_cyl_low     : int = 0
var ide_reg_cyl_high    : int = 0
var ide_reg_drive_head  : int = 0
var ide_sector_buffer   : PackedByteArray
var ide_buffer_index    : int = 0
var ide_is_writing_op   : bool = false
var ide_disk_file_path  : String = "user://pc10_hdd_image.bin"
var ide_disk_file       : FileAccess

func pc10_ide_controller_init() -> void:
	ide_sector_buffer = PackedByteArray()
	ide_sector_buffer.resize(IDE_SECTOR_SIZE)
	ide_sector_buffer.fill(0)
	if not FileAccess.file_exists(ide_disk_file_path):
		var f : FileAccess = FileAccess.open(ide_disk_file_path, FileAccess.WRITE)
		if f:
			var blank : PackedByteArray = PackedByteArray()
			blank.resize(IDE_SECTOR_SIZE)
			for _i in range(64):
				f.store_buffer(blank)
			f.close()
	ide_reg_status = IDE_STATUS_DRDY

func write_hardware_port_ide(port: int, value: int) -> void:
	match port:
		0x1F2: ide_reg_sector_count = value
		0x1F3: ide_reg_sector_num   = value
		0x1F4: ide_reg_cyl_low      = value
		0x1F5: ide_reg_cyl_high     = value
		0x1F6: ide_reg_drive_head   = value
		0x1F7: pc10_ide_execute_command(value)
		0x1F0:
			if ide_is_writing_op and ide_buffer_index < IDE_SECTOR_SIZE:
				ide_sector_buffer[ide_buffer_index] = value & 0xFF
				ide_buffer_index += 1
				if ide_buffer_index >= IDE_SECTOR_SIZE:
					pc10_ide_finalize_sector_write()

func read_hardware_port_ide(port: int) -> int:
	if port == 0x1F7:
		return ide_reg_status
	if port == 0x1F1:
		return ide_reg_error
	if port == 0x1F0:
		if not ide_is_writing_op and ide_buffer_index < IDE_SECTOR_SIZE:
			var b : int = ide_sector_buffer[ide_buffer_index]
			ide_buffer_index += 1
			if ide_buffer_index >= IDE_SECTOR_SIZE:
				pc10_ide_finalize_sector_read()
			return b
	return 0

func pc10_ide_execute_command(cmd: int) -> void:
	ide_reg_status |= IDE_STATUS_BSY
	ide_reg_status &= ~IDE_STATUS_DRQ

	var head     : int = ide_reg_drive_head & 0x0F
	var cylinder : int = ide_reg_cyl_low | (ide_reg_cyl_high << 8)
	var sector   : int = ide_reg_sector_num
	var abs_logical_sector  : int = ((cylinder * IDE_DISK_HEADS + head) * IDE_DISK_SECTORS) + (sector - 1)
	var physical_file_offset : int = abs_logical_sector * IDE_SECTOR_SIZE

	ide_disk_file = FileAccess.open(ide_disk_file_path, FileAccess.READ_WRITE)
	if not ide_disk_file:
		ide_reg_status = IDE_STATUS_DRDY | IDE_STATUS_ERR
		ide_reg_error  = 0x04
		return
	ide_disk_file.seek(physical_file_offset)

	match cmd:
		IDE_CMD_DISK_READ:
			ide_is_writing_op = false
			ide_sector_buffer = ide_disk_file.get_buffer(IDE_SECTOR_SIZE)
			ide_disk_file.close()
			ide_buffer_index = 0
			ide_reg_status &= ~IDE_STATUS_BSY
			ide_reg_status |= IDE_STATUS_DRQ
		IDE_CMD_DISK_WRITE:
			ide_is_writing_op = true
			ide_disk_file.close()
			ide_buffer_index = 0
			ide_reg_status &= ~IDE_STATUS_BSY
			ide_reg_status |= IDE_STATUS_DRQ
		_:
			ide_reg_status = IDE_STATUS_DRDY | IDE_STATUS_ERR
			ide_reg_error  = 0x04
			ide_disk_file.close()

func pc10_ide_finalize_sector_read() -> void:
	ide_reg_sector_count -= 1
	if ide_reg_sector_count > 0:
		ide_reg_sector_num += 1
		if ide_reg_sector_num > IDE_DISK_SECTORS:
			ide_reg_sector_num = 1
			ide_reg_drive_head += 1
			if (ide_reg_drive_head & 0x0F) >= IDE_DISK_HEADS:
				ide_reg_drive_head = (ide_reg_drive_head & 0xF0)
				var cyl : int = ide_reg_cyl_low | (ide_reg_cyl_high << 8)
				cyl += 1
				ide_reg_cyl_low  = cyl & 0xFF
				ide_reg_cyl_high = (cyl >> 8) & 0xFF
		pc10_ide_execute_command(IDE_CMD_DISK_READ)
	else:
		ide_reg_status = IDE_STATUS_DRDY
		ide_buffer_index = 0

func pc10_ide_finalize_sector_write() -> void:
	ide_reg_status |= IDE_STATUS_BSY
	ide_reg_status &= ~IDE_STATUS_DRQ

	var head     : int = ide_reg_drive_head & 0x0F
	var cylinder : int = ide_reg_cyl_low | (ide_reg_cyl_high << 8)
	var sector   : int = ide_reg_sector_num
	var abs_logical_sector  : int = ((cylinder * IDE_DISK_HEADS + head) * IDE_DISK_SECTORS) + (sector - 1)
	var physical_file_offset : int = abs_logical_sector * IDE_SECTOR_SIZE

	ide_disk_file = FileAccess.open(ide_disk_file_path, FileAccess.READ_WRITE)
	if ide_disk_file:
		ide_disk_file.seek(physical_file_offset)
		ide_disk_file.store_buffer(ide_sector_buffer)
		ide_disk_file.close()

	ide_reg_sector_count -= 1
	if ide_reg_sector_count > 0:
		ide_reg_sector_num += 1
		ide_buffer_index = 0
		ide_reg_status &= ~IDE_STATUS_BSY
		ide_reg_status |= IDE_STATUS_DRQ
	else:
		ide_reg_status = IDE_STATUS_DRDY
		ide_buffer_index = 0

const PC10_ACCOUNTS_DB_PATH : String = "user://pc10_accounts.db"
var pc10_accounts     : Dictionary = {}
var pc10_current_user : String = ""

func pc10_account_manager_init() -> void:
	pc10_accounts_load()

func pc10_accounts_load() -> void:
	if FileAccess.file_exists(PC10_ACCOUNTS_DB_PATH):
		var file : FileAccess = FileAccess.open(PC10_ACCOUNTS_DB_PATH, FileAccess.READ)
		var json_string : String = file.get_as_text()
		file.close()
		var json : JSON = JSON.new()
		var error : int = json.parse(json_string)
		if error == OK and json.data is Dictionary:
			pc10_accounts = json.data
			print("[AUTH] База данных загружена. Аккаунтов: %d" % pc10_accounts.size())
		else:
			pc10_accounts = {}
	else:
		pc10_accounts = {}
		pc10_accounts_save()

func pc10_accounts_save() -> void:
	var file : FileAccess = FileAccess.open(PC10_ACCOUNTS_DB_PATH, FileAccess.WRITE)
	if file:
		file.store_string(JSON.stringify(pc10_accounts))
		file.close()

func pc10_register_user(username: String, password: String) -> bool:
	if pc10_accounts.has(username):
		return false
	var ctx : HashingContext = HashingContext.new()
	ctx.start(HashingContext.HASH_SHA256)
	ctx.update(password.to_utf8_buffer())
	var hash_result : String = ctx.finish().hex_encode()
	pc10_accounts[username] = {
		"password_hash": hash_result,
		"created_at": Time.get_unix_time_from_system(),
		"last_login": 0
	}
	pc10_accounts_save()
	print("[AUTH] Аккаунт '%s' успешно создан!" % username)
	return true

func pc10_login_user(username: String, password: String) -> bool:
	if not pc10_accounts.has(username):
		return false
	var ctx : HashingContext = HashingContext.new()
	ctx.start(HashingContext.HASH_SHA256)
	ctx.update(password.to_utf8_buffer())
	var input_hash : String = ctx.finish().hex_encode()
	if pc10_accounts[username]["password_hash"] == input_hash:
		pc10_current_user = username
		pc10_accounts[username]["last_login"] = Time.get_unix_time_from_system()
		pc10_accounts_save()
		print("[AUTH] Пользователь '%s' авторизован." % username)
		return true
	return false

func pc10_logout_user() -> void:
	print("[AUTH] Пользователь '%s' вышел из системы." % pc10_current_user)
	pc10_current_user = ""

# Кастомная подфункция INT 21h, AH=0xEE — авторизация через DOS API
# (DS:DX -> структура: [Имя пользователя, 20 байт][Пароль, 20 байт])
func pc10_int21h_ee_login() -> void:
	var user_ptr : int = (cpu_segs["DS"] << 4) + (cpu_regs["DX"] & 0xFFFF)
	var username : String = pc10_read_cstring_from_memory(user_ptr)
	var password : String = pc10_read_cstring_from_memory(user_ptr + 20)
	if pc10_login_user(username, password):
		cpu_regs["AX"] = 0x0001
	else:
		cpu_regs["AX"] = 0x0000

func pc10_read_cstring_from_memory(addr: int) -> String:
	var s : String = ""
	var a : int = addr
	while a < RAM_SIZE and memory[a] != 0:
		s += char(memory[a])
		a += 1
	return s

# ══════════════════════════════════════════════════════════════
#  БЛОК 20 — PC10: РАСШИРЕННЫЙ ДЕКОДЕР ОПКОДОВ (0x0F-группа,
#  PUSHA/POPA, IMUL imm16/imm8, INSB/INSW/OUTSB/OUTSW, DAA/AAS,
#  SHLD/SHRD, PUSH/POP FS/GS). Эти опкоды переводятся через
#  основной cpu_exec(): если встречен 0x0F, читается второй байт
#  и передаётся в pc10_exec_0f(); прочие новые опкоды (0x60/0x61/
#  0x69/0x6B/0x6C-0x6F/0x27/0x3F) тоже маршрутизируются сюда из
#  cpu_exec() через pc10_try_exec_extra_opcode().
# ══════════════════════════════════════════════════════════════

func pc10_try_exec_extra_opcode(op: int) -> bool:
	match op:
		0x60:   # PUSHA
			var temp_sp : int = cpu_regs["SP"]
			cpu_push(cpu_regs["AX"]); cpu_push(cpu_regs["CX"])
			cpu_push(cpu_regs["DX"]); cpu_push(cpu_regs["BX"])
			cpu_push(temp_sp);        cpu_push(cpu_regs["BP"])
			cpu_push(cpu_regs["SI"]); cpu_push(cpu_regs["DI"])
			return true
		0x61:   # POPA
			cpu_regs["DI"] = cpu_pop(); cpu_regs["SI"] = cpu_pop()
			cpu_regs["BP"] = cpu_pop()
			var _discard_sp : int = cpu_pop()
			cpu_regs["BX"] = cpu_pop(); cpu_regs["DX"] = cpu_pop()
			cpu_regs["CX"] = cpu_pop(); cpu_regs["AX"] = cpu_pop()
			return true
		0x69:   # IMUL r16, r/m16, imm16
			var m : Dictionary = cpu_decode_modrm(true)
			var src : int = cpu_get_rm(m)
			var imm : int = cpu_fetch_w()
			var res : int = pc10_calc_imul_signed(src, imm)
			cpu_set_reg(m, res & 0xFFFF)
			return true
		0x6B:   # IMUL r16, r/m16, imm8 (sign-extended)
			var m2 : Dictionary = cpu_decode_modrm(true)
			var src2 : int = cpu_get_rm(m2)
			var imm8 : int = cpu_fetch_sb()
			var res2 : int = pc10_calc_imul_signed(src2, imm8 & 0xFFFF if imm8 >= 0 else (imm8 + 0x10000))
			cpu_set_reg(m2, res2 & 0xFFFF)
			return true
		0x6C:   # INSB
			var data : int = _in_port(cpu_regs["DX"])
			cpu_wb("ES", cpu_regs["DI"], data & 0xFF)
			cpu_regs["DI"] = (cpu_regs["DI"] + (-1 if cpu_flags["DF"] else 1)) & 0xFFFF
			return true
		0x6D:   # INSW
			var data2 : int = _in_port(cpu_regs["DX"])
			cpu_ww("ES", cpu_regs["DI"], data2 & 0xFFFF)
			cpu_regs["DI"] = (cpu_regs["DI"] + (-2 if cpu_flags["DF"] else 2)) & 0xFFFF
			return true
		0x6E:   # OUTSB
			var outb : int = cpu_rb("DS", cpu_regs["SI"])
			_out_port(cpu_regs["DX"], outb)
			cpu_regs["SI"] = (cpu_regs["SI"] + (-1 if cpu_flags["DF"] else 1)) & 0xFFFF
			return true
		0x6F:   # OUTSW
			var outw : int = cpu_rw("DS", cpu_regs["SI"])
			_out_port(cpu_regs["DX"], outw)
			cpu_regs["SI"] = (cpu_regs["SI"] + (-2 if cpu_flags["DF"] else 2)) & 0xFFFF
			return true
		0x27:   # DAA — Decimal Adjust AL after Addition
			pc10_op_daa()
			return true
		0x3F:   # AAS — ASCII Adjust AL after Subtraction
			pc10_op_aas()
			return true
		0x0F:   # двухбайтовая группа опкодов 386+
			var sub_op : int = cpu_fetch_b()
			pc10_exec_0f(sub_op)
			return true
	return false

func pc10_calc_imul_signed(a: int, b: int) -> int:
	var sa : int = a - 0x10000 if a & 0x8000 else a
	var sb : int = b - 0x10000 if b & 0x8000 else b
	var res : int = sa * sb
	var overflow : bool = (res > 32767 or res < -32768)
	cpu_flags["CF"] = 1 if overflow else 0
	cpu_flags["OF"] = cpu_flags["CF"]
	return res

func pc10_op_daa() -> void:
	var al : int = cpu_regs["AX"] & 0xFF
	var old_al : int = al
	var old_cf : int = cpu_flags["CF"]
	cpu_flags["CF"] = 0
	if (al & 0x0F) > 9 or cpu_flags["AF"] == 1:
		al = (al + 6) & 0xFF
		cpu_flags["AF"] = 1
	else:
		cpu_flags["AF"] = 0
	if old_al > 0x99 or old_cf == 1:
		al = (al + 0x60) & 0xFF
		cpu_flags["CF"] = 1
	cpu_regs["AX"] = (cpu_regs["AX"] & 0xFF00) | al
	cpu_upd_flags(al, 8)

func pc10_op_aas() -> void:
	var al : int = cpu_regs["AX"] & 0xFF
	var ah : int = (cpu_regs["AX"] >> 8) & 0xFF
	if (al & 0x0F) > 9 or cpu_flags["AF"] == 1:
		al = (al - 6) & 0xFF
		ah = (ah - 1) & 0xFF
		cpu_flags["AF"] = 1
		cpu_flags["CF"] = 1
	else:
		cpu_flags["AF"] = 0
		cpu_flags["CF"] = 0
	al = al & 0x0F
	cpu_regs["AX"] = (ah << 8) | al

func pc10_exec_0f(sub_op: int) -> void:
	if sub_op >= 0x80 and sub_op <= 0x8F:
		var offset16 : int = cpu_fetch_sw()
		if pc10_eval_jcc_condition(sub_op & 0x0F):
			cpu_ip = (cpu_ip + offset16) & 0xFFFF
		return
	match sub_op:
		0xA0:   # PUSH FS
			cpu_push(pc10_seg_fs)
		0xA1:   # POP FS
			pc10_seg_fs = cpu_pop()
		0xA8:   # PUSH GS
			cpu_push(pc10_seg_gs)
		0xA9:   # POP GS
			pc10_seg_gs = cpu_pop()
		0xA5:   # SHLD r/m16, reg16, CL
			var m : Dictionary = cpu_decode_modrm(true)
			var val : int = cpu_get_rm(m)
			var reg_val : int = cpu_get_reg(m)
			var shift : int = cpu_regs["CX"] & 0x1F
			if shift > 0:
				var res : int = ((val << shift) | (reg_val >> (16 - shift))) & 0xFFFF
				cpu_set_rm(m, res)
				cpu_upd_flags(res, 16)
		0xAC:   # SHRD r/m16, reg16, CL
			var m2 : Dictionary = cpu_decode_modrm(true)
			var val2 : int = cpu_get_rm(m2)
			var reg_val2 : int = cpu_get_reg(m2)
			var shift2 : int = cpu_regs["CX"] & 0x1F
			if shift2 > 0:
				var res2 : int = ((val2 >> shift2) | (reg_val2 << (16 - shift2))) & 0xFFFF
				cpu_set_rm(m2, res2)
				cpu_upd_flags(res2, 16)
		_:
			_ps_print("[CPU ERR] Неизвестный опкод 0x0F 0x%02X\n" % sub_op)

# Теневые сегментные регистры FS/GS (386+), не входят в базовый cpu_segs PC9
var pc10_seg_fs : int = 0
var pc10_seg_gs : int = 0

func pc10_eval_jcc_condition(jtype: int) -> bool:
	var f : Dictionary = cpu_flags
	match jtype:
		0x0: return f["OF"] == 1
		0x1: return f["OF"] == 0
		0x2: return f["CF"] == 1
		0x3: return f["CF"] == 0
		0x4: return f["ZF"] == 1
		0x5: return f["ZF"] == 0
		0x6: return f["CF"] == 1 or f["ZF"] == 1
		0x7: return f["CF"] == 0 and f["ZF"] == 0
		0x8: return f["SF"] == 1
		0x9: return f["SF"] == 0
		0xA: return f["PF"] == 1
		0xB: return f["PF"] == 0
		0xC: return f["SF"] != f["OF"]
		0xD: return f["SF"] == f["OF"]
		0xE: return f["ZF"] == 1 or f["SF"] != f["OF"]
		0xF: return f["ZF"] == 0 and f["SF"] == f["OF"]
	return false

func pc10_test_ide_controller() -> void:
	_ps_print("\n--- [PC10 IDE CONTROLLER TEST] START ---\n")
	var segment : int = 0x0500
	var offset  : int = 0x0000
	var secret_message : String = "PC10_SECURE_DISK_SECTOR_TEST"
	var ascii_bytes : PackedByteArray = secret_message.to_ascii_buffer()

	for i in range(ascii_bytes.size()):
		var addr : int = (segment << 4) + offset + i
		if addr < RAM_SIZE:
			memory[addr] = ascii_bytes[i]

	write_hardware_port_ide(0x1F2, 1)
	write_hardware_port_ide(0x1F3, 5)
	write_hardware_port_ide(0x1F4, 0)
	write_hardware_port_ide(0x1F5, 0)
	write_hardware_port_ide(0x1F6, 0xA1)
	write_hardware_port_ide(0x1F7, IDE_CMD_DISK_WRITE)

	if (ide_reg_status & IDE_STATUS_DRQ) != 0:
		for i in range(IDE_SECTOR_SIZE):
			var data_byte : int = 0
			if i < ascii_bytes.size():
				var addr2 : int = (segment << 4) + offset + i
				data_byte = memory[addr2] if addr2 < RAM_SIZE else 0
			write_hardware_port_ide(0x1F0, data_byte)

	write_hardware_port_ide(0x1F2, 1)
	write_hardware_port_ide(0x1F3, 5)
	write_hardware_port_ide(0x1F6, 0xA1)
	write_hardware_port_ide(0x1F7, IDE_CMD_DISK_READ)

	var read_buffer : PackedByteArray = PackedByteArray()
	if (ide_reg_status & IDE_STATUS_DRQ) != 0:
		for i in range(ascii_bytes.size()):
			read_buffer.append(read_hardware_port_ide(0x1F0))

	var restored_string : String = read_buffer.get_string_from_ascii()
	_ps_print("[IDE Test] Считано из портов: %s\n" % restored_string)
